22: void wkt_set_clock(wkt_clock_t clock)
23: {
24:         if (clock == WKT_IRC)
25:                 WKT_CTRL &= ~WKT_CTRL_CLKSEL;
26:         else if (clock == WKT_LOW_POWER_CLOCK)
27:                 WKT_CTRL |= WKT_CTRL_CLKSEL;
28: }

30: int wkt_get_status(int status)
31: {
32:         return WKT_CTRL & status;
33: }
34: 
35: void wkt_clear_status(int status)
36: {
37:         WKT_CTRL |= status;
38: }

42:         WKT_COUNT = count;
43: }
