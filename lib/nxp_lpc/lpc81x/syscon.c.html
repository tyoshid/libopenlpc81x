033: static int peripheral_to_reset(int peripheral)
034: {
035:         int reset;
036:         int i;
037:         static const int peripheral_reset[] = {
038:                 0,
039:                 0,
040:                 0,
041:                 SYSCON_PRESETCTRL_FLASH_RST_N,
042:                 0,
043:                 SYSCON_PRESETCTRL_I2C_RST_N,
044:                 SYSCON_PRESETCTRL_GPIO_RST_N,
045:                 0,
046:                 SYSCON_PRESETCTRL_SCT_RST_N,
047:                 SYSCON_PRESETCTRL_WKT_RST_N,
048:                 SYSCON_PRESETCTRL_MRT_RST_N,
049:                 SYSCON_PRESETCTRL_SPI0_RST_N,
050:                 SYSCON_PRESETCTRL_SPI1_RST_N,
051:                 0,
052:                 SYSCON_PRESETCTRL_UART0_RST_N,
053:                 SYSCON_PRESETCTRL_UART1_RST_N,
054:                 SYSCON_PRESETCTRL_UART2_RST_N,
055:                 0,
056:                 0,
057:                 SYSCON_PRESETCTRL_ACMP_RST_N,
058:                 SYSCON_PRESETCTRL_UARTFRG_RST_N
059:         };
060: 
061:         reset = 0;
062:         for (i = 0; i < (int)(sizeof(peripheral_reset) / sizeof(int)); i++) {
063:                 if (peripheral & (1 << i))
064:                         reset |= peripheral_reset[i];
065:         }
066:         return reset;
067: }

079: void syscon_enable_pll(enum syscon_osc source, int m, int p)
080: {
081:         int osc;
082:         int psel;
083:         int msel;
084: 
085:         switch (source) {
086:         case SYSCON_IRC:
087:                 osc = SYSCON_SYSPLLCLKSEL_SEL_IRC;
088:                 break;
089:         case SYSCON_XTAL:
090:                 osc = SYSCON_SYSPLLCLKSEL_SEL_SYSOSC;
091:                 break;
092:         case SYSCON_CLKIN:
093:                 osc = SYSCON_SYSPLLCLKSEL_SEL_CLKIN;
094:                 break;
095:         default:
096:                 return;
097:         }
098: 
099:         for (psel = 0; psel < 4 && p != 1 << psel; psel++)
100:                 ;
101:         if (psel >= 4)
102:                 return;
103: 
104:         if (m < 1 || m > 32)
105:                 return;
106:         msel = m - 1;
107: 
108:         SYSCON_PDRUNCFG &= ~SYSCON_PDRUNCFG_SYSPLL_PD;
109: 
110:         SYSCON_SYSPLLCLKSEL = osc;
111: 
112:         SYSCON_SYSPLLCLKUEN = 0;
113:         SYSCON_SYSPLLCLKUEN = SYSCON_SYSPLLCLKUEN_ENA;
114: 
115:         SYSCON_SYSPLLCTRL = psel << 5 | msel << 0;
116: 
117:         while (!(SYSCON_SYSPLLSTAT & SYSCON_SYSPLLSTAT_LOCK))
118:                 ;
119: }

166: void syscon_set_system_clock(enum syscon_osc source, int div)
167: {
168:         int osc;
169: 
170:         switch (source) {
171:         case SYSCON_IRC:
172:                 osc = SYSCON_MAINCLKSEL_SEL_IRC;
173:                 break;
174:         case SYSCON_PLL_IN:
175:                 osc = SYSCON_MAINCLKSEL_SEL_PLL_IN;
176:                 break;
177:         case SYSCON_WDT_OSC:
178:                 osc = SYSCON_MAINCLKSEL_SEL_WDT_OSC;
179:                 break;
180:         case SYSCON_PLL_OUT:
181:                 osc = SYSCON_MAINCLKSEL_SEL_PLL_OUT;
182:                 break;
183:         default:
184:                 return;
185:         }
186: 
187:         if (div < 0 || div > 255)
188:                 return;
189: 
190:         SYSCON_MAINCLKSEL = osc;
191: 
192:         SYSCON_MAINCLKUEN = 0;
193:         SYSCON_MAINCLKUEN = SYSCON_MAINCLKUEN_ENA;
194: 
195:         SYSCON_SYSAHBCLKDIV = div;
196: }

208: void syscon_set_usart_clock(int main_clock, int u_pclk)
209: {
210:         int n;
211:         int t;
212:         int mult;
213: 
214:         /*
215:          * U_PCLK = main clock / UARTCLKDIV / (1 + MULT / 256)
216:          * MULT = 256 * (main clock / (UARTCLKDIV * U_PCLK) - 1)
217:          */
218:         if (u_pclk <= 0)
219:                 return;
220:         n = main_clock / u_pclk;
221:         if (n > 255 || n <= 0)
222:                 return;
223:         t = u_pclk * n;
224:         mult = 256 * (main_clock - t) / t;
225: 
226:         SYSCON_UARTCLKDIV = n;
227:         if (mult) {
228:                 SYSCON_UARTFRGDIV = 0xff;
229:                 SYSCON_UARTFRGMULT = mult;
230:         } else {
231:                 SYSCON_UARTFRGDIV = 0;
232:                 SYSCON_UARTFRGMULT = 0;
233:         }
234: }

332: void syscon_select_pins(int offset, int pins)
333: {
334:         int i;
335:         int j;
336: 
337:         j = offset;
338:         if (j >= 8)
339:                 return;
340:         for (i = 0; i < 18; i++) {
341:                 if (pins & (1 << i)) {
342:                         SYSCON_PINTSEL(j++) = i;
343:                         if (j >= 8)
344:                                 break;
345:                 }
346:         }
347: }
