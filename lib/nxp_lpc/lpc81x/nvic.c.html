22: int nvic_get_enabled_irq(int interrupts)
23: {
24:         return NVIC_ISER0 & interrupts;
25: }
26: 
27: void nvic_enable_irq(int interrupts)
28: {
29:         NVIC_ISER0 = interrupts;
30: }
31: 
32: void nvic_disable_irq(int interrupts)
33: {
34:         NVIC_ICER0 = interrupts;
35: }
36: 
37: int nvic_get_pending_irq(int interrupts)
38: {
39:         return NVIC_ISPR0 & interrupts;
40: }
41: 
42: void nvic_set_pending_irq(int interrupts)
43: {
44:         NVIC_ISPR0 = interrupts;
45: }
46: 
47: void nvic_clear_pending_irq(int interrupts)
48: {
49:         NVIC_ICPR0 = interrupts;
50: }
51: 
52: int nvic_get_active_irq(int interrupts)
53: {
54:         return NVIC_IABR0 & interrupts;
55: }
56: 
57: void nvic_set_priority(int irqn, int priority)
58: {
59:         int r;
60: 
61:         r = NVIC_IPR(irqn / 4);
62:         r &= ~(NVIC_PRIO_3 << irqn % 4 * 8);
63:         NVIC_IPR(irqn / 4) = r | priority << irqn % 4 * 8;
64: }
65: 
66: int nvic_get_priority(int irqn)
67: {
68:         int r;
69: 
70:         r = NVIC_IPR(irqn / 4);
71:         return (r >> (irqn % 4 * 8)) & 0xff;
72: }
