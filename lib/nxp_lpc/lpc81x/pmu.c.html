20: #include <pmu.h>
21: 
22: void pmu_set_power_mode(pmu_power_mode_t pm)
23: {
24:         int r;
25: 
26:         r = PMU_PCON;
27:         r &= ~3;
28:         PMU_PCON = r | pm;
29: }
30: 
31: pmu_power_mode_t pmu_get_power_mode(void)
32: {
33:         return PMU_PCON & 3;
34: }
35: 
36: void pmu_prevent_deep_power_down(void)
37: {
38:         PMU_PCON |= PMU_PCON_NODPD;
39: }
40: 
41: int pmu_get_flag(int flag)
42: {
43:         return PMU_PCON & flag;
44: }
45: 
46: void pmu_clear_flag(int flag)
47: {
48:         PMU_PCON |= flag & (PMU_PCON_SLEEPFLAG | PMU_PCON_DPDFLAG);
49: }
50: 
51: void pmu_enable_low_power_osc(bool deep_power_down)
52: {
53:         int r;
54: 
55:         r = PMU_DPDCTRL;
56:         if (deep_power_down)
57:                 r |= PMU_DPDCTRL_LPOSCDPDEN;
58:         else
59:                 r &= ~PMU_DPDCTRL_LPOSCDPDEN;
60:         PMU_DPDCTRL = r | PMU_DPDCTRL_LPOSCEN;
61: }
62: 
63: void pmu_disable_low_power_osc(void)
64: {
65:         PMU_DPDCTRL &= ~(PMU_DPDCTRL_LPOSCEN | PMU_DPDCTRL_LPOSCDPDEN);
66: }
67: 
68: void pmu_disable_wakeup_pin(void)
69: {
70:         int r;
71: 
72:         r = PMU_DPDCTRL;
73:         r &= ~PMU_DPDCTRL_WAKEUPHYS;
74:         PMU_DPDCTRL = r | PMU_DPDCTRL_WAKEPAD_DISABLE;
75: }
76: 
77: void pmu_enable_wakeup_pin(bool hysteresis)
78: {
79:         int r;
80: 
81:         r = PMU_DPDCTRL;
82:         if (hysteresis)
83:                 r |= PMU_DPDCTRL_WAKEUPHYS;
84:         else
85:                 r &= ~PMU_DPDCTRL_WAKEUPHYS;
86:         PMU_DPDCTRL = r & ~PMU_DPDCTRL_WAKEPAD_DISABLE;
87: }
