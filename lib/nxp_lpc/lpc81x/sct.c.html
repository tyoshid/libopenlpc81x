138: void sct_set_counter_mode(sct_counter_t mode)
139: {
140:         int r;
141: 
142:         r = SCT_CTRL;
143:         switch (mode) {
144:         case SCT_COUNT_UP:
145:                 r &= ~(SCT_CTRL_BIDIR_L | SCT_CTRL_DOWN_L);
146:                 break;
147:         case SCT_COUNT_BIDIR_UP:
148:                 r &= ~SCT_CTRL_DOWN_L;
149:                 r |= SCT_CTRL_BIDIR_L;
150:                 break;
151:         case SCT_COUNT_BIDIR_DOWN:
152:                 r |= SCT_CTRL_BIDIR_L | SCT_CTRL_DOWN_L;
153:                 break;
154:         default:
155:                 break;
156:         }
157:         SCT_CTRL = r;
158: }

369: void sct_enable_output_counter_control(int output)
370: {
371:         int i;
372:         int r;
373: 
374:         r = SCT_OUTPUTDIRCTRL;
375:         for (i = 0; i < 4; i++) {
376:                 if (output & (1 << i)) {
377:                         r |= 1 << i * 2;
378:                         r &= ~(2 << i * 2);
379:                 }
380:         }
381:         SCT_OUTPUTDIRCTRL = r;
382: }

427: void sct_enable_interrupt(int interrupt)
428: {
429:         if (interrupt & 0x3f)
430:                 SCT_EVEN |= interrupt & 0x3f;
431:         if (interrupt & 0xf00)
432:                 SCT_CONEN |= (interrupt >> 8) & 0xf;
433: }

435: void sct_disable_interrupt(int interrupt)
436: {
437:         if (interrupt & 0x3f)
438:                 SCT_EVEN &= ~(interrupt & 0x3f);
439:         if (interrupt & 0xf00)
440:                 SCT_CONEN &= ~((interrupt >> 8) & 0xf);
441: }

443: int sct_get_interrupt_mask(int interrupt)
444: {
445:         int r = 0;
446: 
447:         if (interrupt & 0x3f)
448:                 r = SCT_EVEN & (interrupt & 0x3f);
449:         if (interrupt & 0xf00)
450:                 r |= (SCT_CONEN << 8) & (interrupt & 0xf00);
451:         return r;
452: }

546: void sct_setup_event(int ev, int match, int io, int op, int action, int state)
547: {
548:         if (action & SCT_EV_LIMIT)
549:                 SCT_LIMIT |= 1 << ev;
550:         if (action & SCT_EV_LIMIT_H)
551:                 SCT_LIMIT_H |= 1 << ev;
552: 
553:         if (action & SCT_EV_HALT)
554:                 SCT_HALT |= 1 << ev;
555:         if (action & SCT_EV_HALT_H)
556:                 SCT_HALT_H |= 1 << ev;
557: 
558:         if (action & SCT_EV_STOP)
559:                 SCT_STOP |= 1 << ev;
560:         if (action & SCT_EV_STOP_H)
561:                 SCT_STOP_H |= 1 << ev;
562: 
563:         if (action & SCT_EV_START)
564:                 SCT_START |= 1 << ev;
565:         if (action & SCT_EV_START_H)
566:                 SCT_START_H |= 1 << ev;
567: 
568:         if (action & SCT_EV_INT)
569:                 SCT_EVEN |= 1 << ev;
570: 
571:         if (action & SCT_EV_OUT0_SET)
572:                 SCT_OUT0_SET |= 1 << ev;
573:         if (action & SCT_EV_OUT0_CLR)
574:                 SCT_OUT0_CLR |= 1 << ev;
575:         if (action & SCT_EV_OUT1_SET)
576:                 SCT_OUT1_SET |= 1 << ev;
577:         if (action & SCT_EV_OUT1_CLR)
578:                 SCT_OUT1_CLR |= 1 << ev;
579:         if (action & SCT_EV_OUT2_SET)
580:                 SCT_OUT2_SET |= 1 << ev;
581:         if (action & SCT_EV_OUT2_CLR)
582:                 SCT_OUT2_CLR |= 1 << ev;
583:         if (action & SCT_EV_OUT3_SET)
584:                 SCT_OUT3_SET |= 1 << ev;
585:         if (action & SCT_EV_OUT3_CLR)
586:                 SCT_OUT3_CLR |= 1 << ev;
587: 
588:         if (action & SCT_EV_CAP0)
589:                 SCT_CAP0 |= 1 << ev;
590:         if (action & SCT_EV_CAP0_H)
591:                 SCT_CAP0_H |= 1 << ev;
592:         if (action & SCT_EV_CAP1)
593:                 SCT_CAP1 |= 1 << ev;
594:         if (action & SCT_EV_CAP1_H)
595:                 SCT_CAP1_H |= 1 << ev;
596:         if (action & SCT_EV_CAP2)
597:                 SCT_CAP2 |= 1 << ev;
598:         if (action & SCT_EV_CAP2_H)
599:                 SCT_CAP2_H |= 1 << ev;
600:         if (action & SCT_EV_CAP3)
601:                 SCT_CAP3 |= 1 << ev;
602:         if (action & SCT_EV_CAP3_H)
603:                 SCT_CAP3_H |= 1 << ev;
604:         if (action & SCT_EV_CAP4)
605:                 SCT_CAP4 |= 1 << ev;
606:         if (action & SCT_EV_CAP4_H)
607:                 SCT_CAP4_H |= 1 << ev;
608: 
609:         SCT_EV_CTRL(ev) = (match & 0xf) | (io & 0xf) << 6 |
610:                 (state & 0x1f) << 15 |
611:                 ((action & SCT_EV_STATE_LOAD) ? SCT_EV_CTRL_STATELD : 0) |
612:                 (op & ~0xfc3cf);
613:         SCT_EV_STATE(ev) = op & 3;
614: }
