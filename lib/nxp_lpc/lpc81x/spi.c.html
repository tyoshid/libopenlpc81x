020: #include <spi.h>
021: 
022: static int base_addr(spi_t spi)
023: {
024:         switch (spi) {
025:         case SPI0:
026:                 return SPI0_BASE;
027:         case SPI1:
028:                 return SPI1_BASE;
029:         default:
030:                 break;
031:         }
032:         return 0;
033: }
034: 
035: void spi_init_master(spi_t spi, int config, int clkdiv, int pre_delay,
036:                      int post_delay, int frame_delay, int transfer_delay)
037: {
038:         int base;
039: 
040:         base = base_addr(spi);
041:         SPI_DIV(base) = (clkdiv - 1) & 0xffff;
042:         SPI_DLY(base) = (pre_delay & 0xf) | ((post_delay & 0xf) << 4) |
043:                 ((frame_delay & 0xf) << 8) | ((transfer_delay & 0xf) << 12);
044:         SPI_CFG(base) = (config & 0x1b8) | SPI_CFG_MASTER | SPI_CFG_ENABLE;
045: }
046: 
047: void spi_init_slave(spi_t spi, int config)
048: {
049:         int base;
050: 
051:         base = base_addr(spi);
052:         SPI_CFG(base) = (config & 0x1b8) | SPI_CFG_ENABLE;
053: }
054: 
055: void spi_enable(spi_t spi)
056: {
057:         int base;
058:         int r;
059: 
060:         base = base_addr(spi);
061:         r = SPI_CFG(base);
062:         SPI_CFG(base) = (r & 0x1bc) | SPI_CFG_ENABLE;
063: }
064: 
065: void spi_disable(spi_t spi)
066: {
067:         int base;
068:         int r;
069: 
070:         base = base_addr(spi);
071:         r = SPI_CFG(base);
072:         SPI_CFG(base) = r & 0x1bc;
073: }
074: 
075: void spi_enable_loop_back(spi_t spi)
076: {
077:         int base;
078:         int r;
079: 
080:         base = base_addr(spi);
081:         r = SPI_CFG(base);
082:         SPI_CFG(base) = (r & 0x13d) | SPI_CFG_LOOP;
083: }
084: 
085: void spi_disable_loop_back(spi_t spi)
086: {
087:         int base;
088:         int r;
089: 
090:         base = base_addr(spi);
091:         r = SPI_CFG(base);
092:         SPI_CFG(base) = r & 0x13d;
093: }
094: 
095: void spi_send(spi_t spi, int data)
096: {
097:         SPI_TXDAT(base_addr(spi)) = data & 0xffff;
098: }
099: 
100: void spi_set_tx_control(spi_t spi, int len, int control)
101: {
102:         SPI_TXCTL(base_addr(spi)) = (((len - 1) & 0xf) << 24) |
103:                 (control & 0x710000);
104: }
105: 
106: void spi_send_control(spi_t spi, int data, int len, int control)
107: {
108:         SPI_TXDATCTL(base_addr(spi)) = (((len - 1) & 0xf) << 24) |
109:                 (control & 0x710000) | (data & 0xffff);
110: }
111: 
112: int spi_recv(spi_t spi)
113: {
114:         return SPI_RXDAT(base_addr(spi)) & 0x11ffff;
115: }
116: 
117: void spi_send_blocking(spi_t spi, int data)
118: {
119:         int base;
120: 
121:         base = base_addr(spi);
122:         while (!(SPI_STAT(base) & SPI_STAT_TXRDY))
123:                 ;
124:         SPI_TXDAT(base) = data & 0xffff;
125: }
126: 
127: void spi_send_control_blocking(spi_t spi, int data, int len, int control)
128: {
129:         int base;
130: 
131:         base = base_addr(spi);
132:         while (!(SPI_STAT(base) & SPI_STAT_TXRDY))
133:                 ;
134:         SPI_TXDATCTL(base) = (((len - 1) & 0xf) << 24) |
135:                 (control & 0x710000) | (data & 0xffff);
136: }
137: 
138: int spi_recv_blocking(spi_t spi)
139: {
140:         int base;
141: 
142:         base = base_addr(spi);
143:         while (!(SPI_STAT(base) & SPI_STAT_RXRDY))
144:                 ;
145:         return SPI_RXDAT(base) & 0x11ffff;
146: }
147: 
148: void spi_enable_interrupt(spi_t spi, int interrupt)
149: {
150:         SPI_INTENSET(base_addr(spi)) = interrupt & 0x3f;
151: }
152: 
153: void spi_disable_interrupt(spi_t spi, int interrupt)
154: {
155:         SPI_INTENCLR(base_addr(spi)) = interrupt & 0x3f;
156: }
157: 
158: int spi_get_interrupt_mask(spi_t spi, int interrupt)
159: {
160:         return  SPI_INTENSET(base_addr(spi)) & interrupt;
161: }
162: 
163: int spi_get_interrupt_status(spi_t spi, int interrupt)
164: {
165:         return  SPI_INTSTAT(base_addr(spi)) & interrupt;
166: }

168: int spi_get_status(spi_t spi, int status)
169: {
170:         return  SPI_STAT(base_addr(spi)) & status;
171: }
172: 
173: void spi_clear_status(spi_t spi, int status)
174: {
175:         SPI_STAT(base_addr(spi)) = status & 0x1ff;
176: }
