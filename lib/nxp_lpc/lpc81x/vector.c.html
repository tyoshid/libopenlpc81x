052: void _reset(void)
053: {
054:         int i;
055:         extern char _data_start[];
056:         extern char _data_load_start[];
057:         extern char _data_size[];
058:         extern char _bss_start[];
059:         extern char _bss_size[];
060:         extern int main(void);
061: 
062:         /* Move initialized data (Flash) to RAM */
063:         for (i = 0; i < (int)_data_size; i++)
064:                 _data_start[i] = _data_load_start[i];
065: 
066:         /* Clear BSS. */
067:         for (i = 0; i < (int)_bss_size; i++)
068:                 _bss_start[i] = 0;
069: 
070:         /* Go to user application. */
071:         main();
072: 
073:         while (1)
074:                 ;
075: }
