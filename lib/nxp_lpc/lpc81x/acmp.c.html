20: #include <acmp.h>
21: 
22: void acmp_init(acmp_input_t p, acmp_input_t n, acmp_hys_t hys, bool sync,
23:                acmp_edge_t edge)
24: {
25:         ACMP_CTRL = (hys << 25) | (n << 11) | (p << 8) |
26:                 (sync ? ACMP_CTRL_COMPSA : 0) | (edge << 3);
27: }
28: 
29: void acmp_enable_ladder(bool vddcmp)
30: {
31:         ACMP_LAD = (vddcmp ? ACMP_LAD_LADREF : 0) | ACMP_LAD_LADEN;
32: }
33: 
34: void acmp_disable_ladder(void)
35: {
36:         ACMP_LAD = 0;
37: }
38: 
39: void acmp_set_ladder(int ladder)
40: {
41:         int r;
42: 
43:         r = ACMP_LAD;
44:         r &= ~(0x1f << 1);
45:         ACMP_LAD = r | ((ladder & 0x1f) << 1);
46: }
47: 
48: int acmp_get_status(int status)
49: {
50:         return ACMP_CTRL & status;
51: }
52: 
53: void acmp_clear_interrupt(void)
54: {
55:         int r;
56: 
57:         r = ACMP_CTRL;
58:         ACMP_CTRL = r | ACMP_CTRL_EDGECLR;
59:         ACMP_CTRL = r & ~ACMP_CTRL_EDGECLR;
60: }
