022: void pinint_set_interrupt_mode(pinint_mode_t mode, int pinints)
023: {
024:         switch (mode) {
025:         case PININT_DISABLE:
026:                 PININT_CIENR = pinints;
027:                 PININT_CIENF = pinints;
028:                 break;
029:         case PININT_RISING:
030:                 PININT_ISEL &= ~pinints;
031:                 PININT_SIENR = pinints;
032:                 PININT_CIENF = pinints;
033:                 break;
034:         case PININT_FALLING:
035:                 PININT_ISEL &= ~pinints;
036:                 PININT_CIENR = pinints;
037:                 PININT_SIENF = pinints;
038:                 break;
039:         case PININT_BOTH:
040:                 PININT_ISEL &= ~pinints;
041:                 PININT_SIENR = pinints;
042:                 PININT_SIENF = pinints;
043:                 break;
044:         case PININT_LOW:
045:                 PININT_ISEL |= pinints;
046:                 PININT_CIENF = pinints;
047:                 PININT_SIENR = pinints;
048:                 break;
049:         case PININT_HIGH:
050:                 PININT_ISEL |= pinints;
051:                 PININT_SIENF = pinints;
052:                 PININT_SIENR = pinints;
053:                 break;
054:         default:
055:                 break;
056:         }
057: }
058: 
059: pinint_mode_t pinint_get_interrupt_mode(int pinints)
060: {
061:         if ((int)(PININT_ISEL & pinints) == pinints) {
062:                 /* Level sensitive */
063:                 if ((int)(PININT_IENR & pinints) == pinints) {
064:                         if ((int)(PININT_IENF & pinints) == pinints)
065:                                 return PININT_HIGH;
066:                         else if ((PININT_IENF & pinints) == 0)
067:                                 return PININT_LOW;
068:                         return PININT_DISABLE;
069:                 } else if ((PININT_IENR & pinints) == 0) {
070:                         return PININT_DISABLE;
071:                 }
072:                 return PININT_UNKNOWN;
073:         } else if ((PININT_ISEL & pinints) == 0) {
074:                 /* Edge sensitive */
075:                 if ((int)(PININT_IENR & pinints) == pinints) {
076:                         if ((int)(PININT_IENF & pinints) == pinints)
077:                                 return PININT_BOTH;
078:                         else
079:                                 return PININT_RISING;
080:                 } else if ((PININT_IENR & pinints) == 0) {
081:                         if ((int)(PININT_IENF & pinints) == pinints)
082:                                 return PININT_FALLING;
083:                         else
084:                                 return PININT_DISABLE;
085:                 }
086:         }
087: 
088:         return PININT_UNKNOWN;
089: }

121: void pinint_set_bit_slice(int bitslice, int pinint, pinint_match_t match,
122:                           bool endpoint)
123: {
124:         int r;
125: 
126:         bitslice &= 7;
127: 
128:         r = PININT_PMSRC;
129:         r &= ~(7 << (bitslice * 3 + 8));
130:         PININT_PMSRC = r | ((pinint & 7) << (bitslice * 3 + 8));
131: 
132:         r = PININT_PMCFG;
133:         r &= ~(7 << (bitslice * 3 + 8));
134:         if (bitslice < 7) {
135:                 if (endpoint)
136:                         r |= 1 << bitslice;
137:                 else
138:                         r &= ~(1 << bitslice);
139:         }
140:         PININT_PMCFG = r | (match << (bitslice * 3 + 8));
141: }
