020: #include <i2c.h>
021: 
022: void i2c_set_scl(int div, int high, int low)
023: {
024:         I2C_CLKDIV = (div - 1) & 0xffff;
025:         I2C_MSTTIME = ((high - 2) & 7) << 4 | ((low - 2) & 7);
026: }
027: 
028: void i2c_set_time_out(int to)
029: {
030:         I2C_TIMEOUT = (to - 1) & 0xffff;
031: }
032: 
033: void i2c_enable_master(bool timeout)
034: {
035:         int r;
036: 
037:         r = I2C_CFG;
038:         if (timeout)
039:                 r |= I2C_CFG_TIMEOUTEN;
040:         else
041:                 r &= ~I2C_CFG_TIMEOUTEN;
042:         I2C_CFG = r | I2C_CFG_MSTEN;
043: }
044: 
045: void i2c_disable_master(void)
046: {
047:         I2C_CFG &= ~I2C_CFG_MSTEN;
048: }
049: 
050: void i2c_start(void)
051: {
052:         I2C_MSTCTL = I2C_MSTCTL_MSTSTART;
053: }
054: 
055: void i2c_stop(void)
056: {
057:         I2C_MSTCTL = I2C_MSTCTL_MSTSTOP;
058: }
059: 
060: void i2c_master_continue(void)
061: {
062:         I2C_MSTCTL = I2C_MSTCTL_MSTCONTINUE;
063: }
064: 
065: void i2c_master_send_addr(int addr)
066: {
067:         I2C_MSTDAT = addr & 0xff;
068:         I2C_MSTCTL = I2C_MSTCTL_MSTSTART;
069: }
070: 
071: void i2c_master_send(int data)
072: {
073:         I2C_MSTDAT = data & 0xff;
074:         I2C_MSTCTL = I2C_MSTCTL_MSTCONTINUE;
075: }
076: 
077: int i2c_master_recv(void)
078: {
079:         int r;
080: 
081:         r = I2C_MSTDAT;
082:         I2C_MSTCTL = I2C_MSTCTL_MSTCONTINUE;
083:         return r & 0xff;
084: }
085:

139: void i2c_enable_monitor(bool timeout, bool clkstr)
140: {
141:         int r;
142: 
143:         r = I2C_CFG;
144:         if (timeout)
145:                 r |= I2C_CFG_TIMEOUTEN;
146:         else
147:                 r &= ~I2C_CFG_TIMEOUTEN;
148:         if (clkstr)
149:                 r |= I2C_CFG_MONCLKSTR;
150:         else
151:                 r &= ~I2C_CFG_MONCLKSTR;
152:         I2C_CFG = r | I2C_CFG_MONEN;
153: }

157:         I2C_CFG &= ~I2C_CFG_MONEN;
158: }

170: void i2c_clear_status(int status)
171: {
172:         I2C_STAT = status & (I2C_STAT_SCLTIMEOUT | I2C_STAT_EVENTTIMEOUT |
173:                              I2C_STAT_MONIDLE | I2C_STAT_MONOV |
174:                              I2C_STAT_SLVDESEL | I2C_STAT_MSTSTSTPERR |
175:                              I2C_STAT_MSTARBLOSS);
176: }
