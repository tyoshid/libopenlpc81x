032: /* --- PININT registers ---------------------------------------------------- */
033: /*
034:  * Offset       Register
035:  * 0x000        PININT_ISEL             Pin interrupt mode register
036:  * 0x004        PININT_IENR             Pin interrupt level or rising edge
037:  *                                      interrupt enable register
038:  * 0x008        PININT_SIENR            Pin interrupt level or rising edge
039:  *                                      interrupt set register
040:  * 0x00c        PININT_CIENR            Pin interrupt level or rising edge
041:  *                                      interrupt clear register
042:  * 0x010        PININT_IENF             Pin interrupt active level or falling
043:  *                                      edge interrupt enable register
044:  * 0x014        PININT_SIENF            Pin interrupt active level or falling
045:  *                                      edge interrupt set register
046:  * 0x018        PININT_CIENF            Pin interrupt active level or falling
047:  *                                      edge interrupt clear register
048:  * 0x01c        PININT_RISE             Pin interrupt rising edge register
049:  * 0x020        PININT_FALL             Pin interrupt falling edge register
050:  * 0x024        PININT_IST              Pin interrupt status register
051:  * 0x028        PININT_PMCTRL           Pattern match interrupt control
052:  *                                      register
053:  * 0x02c        PININT_PMSRC            Pattern match interrupt bit-slice
054:  *                                      source register
055:  * 0x030        PININT_PMCFG            Pattern match interrupt bit slice
056:  *                                      configuration register
057:  */
058: 
059: #define PININT_ISEL                     MMIO32(PININT_BASE + 0x000)
060: #define PININT_IENR                     MMIO32(PININT_BASE + 0x004)
061: #define PININT_SIENR                    MMIO32(PININT_BASE + 0x008)
062: #define PININT_CIENR                    MMIO32(PININT_BASE + 0x00c)
063: #define PININT_IENF                     MMIO32(PININT_BASE + 0x010)
064: #define PININT_SIENF                    MMIO32(PININT_BASE + 0x014)
065: #define PININT_CIENF                    MMIO32(PININT_BASE + 0x018)
066: #define PININT_RISE                     MMIO32(PININT_BASE + 0x01c)
067: #define PININT_FALL                     MMIO32(PININT_BASE + 0x020)
068: #define PININT_IST                      MMIO32(PININT_BASE + 0x024)
069: #define PININT_PMCTRL                   MMIO32(PININT_BASE + 0x028)
070: #define PININT_PMSRC                    MMIO32(PININT_BASE + 0x02c)
