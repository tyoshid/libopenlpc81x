32: /* --- SysTick registers --------------------------------------------------- */
33: /*
34:  * Offset       Register
35:  * 0x000        SYST_CSR                System Timer Control and status
36:  *                                      register
37:  * 0x004        SYST_RVR                System Timer Reload value register
38:  * 0x008        SYST_CVR                System Timer Current value register
39:  * 0x00c        SYST_CALIB              System Timer Calibration value register
40:  */
41: 
42: #define SYST_CSR                        MMIO32(STK_BASE + 0x000)
43: #define SYST_RVR                        MMIO32(STK_BASE + 0x004)
44: #define SYST_CVR                        MMIO32(STK_BASE + 0x008)
45: #define SYST_CALIB                      MMIO32(STK_BASE + 0x00c)
46: 
47: /* --- SYST_CSR values ----------------------------------------------------- */
48: 
49: #define SYST_CSR_COUNTFLAG              (1 << 16)
50: #define SYST_CSR_CLKSOURCE              (1 << 2)
51: #define SYST_CSR_TICKINT                (1 << 1)
52: #define SYST_CSR_ENABLE                 (1 << 0)
53: 
54: /* --- SYST_RVR values ----------------------------------------------------- */
55: 
56: /* SYST_RVR[23:0]: RELOAD */
57: 
58: /* --- SYST_CVR values ----------------------------------------------------- */
59: 
60: /* SYST_CVR[23:0]: CURRENT */

62: /* --- SYST_CALIB values --------------------------------------------------- */
63: 
64: #define SYST_CALIB_NOREF                (1 << 31)
65: #define SYST_CALIB_SKEW                 (1 << 30)
