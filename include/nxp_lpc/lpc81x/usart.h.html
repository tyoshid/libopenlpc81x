032: /* --- USART registers ----------------------------------------------------- */
033: /*
034:  * Offset       Register
035:  * 0x000        USART_CFG               USART Configuration register
036:  * 0x004        USART_CTL               USART Control register
037:  * 0x008        USART_STAT              USART Status register
038:  * 0x00c        USART_INTENSET          Interrupt Enable read and Set register
039:  * 0x010        USART_INTENCLR          Interrupt Enable Clear register
040:  * 0x014        USART_RXDAT             Receiver Data register
041:  * 0x018        USART_RXDATSTAT         Receiver Data with Status register
042:  * 0x01c        USART_TXDAT             Transmit Data register
043:  * 0x020        USART_BRG               Baud Rate Generator register
044:  * 0x024        USART_INTSTAT           Interrupt status register
045:  */
