032: /* --- SPIx registers ------------------------------------------------------ */
033: /*
034:  * Offset       Register
035:  * 0x000        SPIx_CFG                SPI Configuration register
036:  * 0x004        SPIx_DLY                SPI Delay register
037:  * 0x008        SPIx_STAT               SPI Status register
038:  * 0x00c        SPIx_INTENSET           SPI Interrupt Enable read Set register
039:  * 0x010        SPIx_INTENCLR           SPI Interrupt Enable Clear register
040:  * 0x014        SPIx_RXDAT              SPI Receiver Data register
041:  * 0x018        SPIx_TXDATCTL           SPI Transmitter Data and Control
042:  *                                      register
043:  * 0x01c        SPIx_TXDAT              SPI Transmitter Data register
044:  * 0x020        SPIx_TXCTL              SPI Transmitter Control register
045:  * 0x024        SPIx_DIV                SPI Divider register
046:  * 0x028        SPIx_INTSTAT            SPI Interrupt Status register
047:  */
