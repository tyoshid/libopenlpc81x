029: #include <mmio.h>

032: /* --- ACMP registers ------------------------------------------------------ */
033: /*
034:  * Offset       Register
035:  * 0x000        ACMP_CTRL               Comparator control register
036:  * 0x004        ACMP_LAD                Voltage ladder register
037:  */
038: 
039: #define ACMP_CTRL                       MMIO32(ACMP_BASE + 0x000)
040: #define ACMP_LAD                        MMIO32(ACMP_BASE + 0x004)
041: 
042: /* --- ACMP_CTRL values ---------------------------------------------------- */
043: 
044: #define ACMP_CTRL_HYS1                  (1 << 26)
045: #define ACMP_CTRL_HYS0                  (1 << 25)
046: #define ACMP_CTRL_COMPEDGE              (1 << 23)
047: #define ACMP_CTRL_COMPSTAT              (1 << 21)
048: #define ACMP_CTRL_EDGECLR               (1 << 20)
049: #define ACMP_CTRL_COMP_VM_SEL2          (1 << 13)
050: #define ACMP_CTRL_COMP_VM_SEL1          (1 << 12)
051: #define ACMP_CTRL_COMP_VM_SEL0          (1 << 11)
052: #define ACMP_CTRL_COMP_VP_SEL2          (1 << 10)
053: #define ACMP_CTRL_COMP_VP_SEL1          (1 << 9)
054: #define ACMP_CTRL_COMP_VP_SEL0          (1 << 8)
055: #define ACMP_CTRL_COMPSA                (1 << 6)
056: #define ACMP_CTRL_EDGESEL1              (1 << 4)
057: #define ACMP_CTRL_EDGESEL0              (1 << 3)
058: 
059: /* HYS */
060: #define ACMP_CTRL_HYS_NONE              (0 << 25)
061: #define ACMP_CTRL_HYS_5MV               (1 << 25)
062: #define ACMP_CTRL_HYS_10MV              (2 << 25)
063: #define ACMP_CTRL_HYS_20MV              (3 << 25)
064: 
065: /* COMP_VM_SEL */
066: #define ACMP_CTRL_COMP_VM_SEL_LADDER    (0 << 11)
067: #define ACMP_CTRL_COMP_VM_SEL_ACMP_I1   (1 << 11)
068: #define ACMP_CTRL_COMP_VM_SEL_ACMP_I2   (2 << 11)
069: #define ACMP_CTRL_COMP_VM_SEL_BANDGAP   (6 << 11)
070: 
071: /* COMP_VP_SEL */
072: #define ACMP_CTRL_COMP_VP_SEL_LADDER    (0 << 8)
073: #define ACMP_CTRL_COMP_VP_SEL_ACMP_I1   (1 << 8)
074: #define ACMP_CTRL_COMP_VP_SEL_ACMP_I2   (2 << 8)
075: #define ACMP_CTRL_COMP_VP_SEL_BANDGAP   (6 << 8)
076: 
077: /* EDGESEL */
078: #define ACMP_CTRL_EDGESEL_FALL          (0 << 3)
079: #define ACMP_CTRL_EDGESEL_RISE          (1 << 3)
080: #define ACMP_CTRL_EDGESEL_BOTH          (2 << 3)
081: 
082: /* --- ACMP_LAD values ----------------------------------------------------- */
083: 
084: #define ACMP_LAD_LADREF                 (1 << 6)
085: /* ACMP_LAD[5:1]: LADSEL */
086: #define ACMP_LAD_LADEN                  (1 << 0)
087: 
088: /* --- Comparator voltage ladder dynamic characteristics  ------------------ */
089: 
090: /* power-up settling time */
091: #define ACMP_TS_PU                              30 /* 30 usec */
092: 
093: /* switching settling time */
094: #define ACMP_TS_SW                              15 /* 15 usec */
095: 
096: /* --- Function prototypes ------------------------------------------------- */
097: 
098: /* Input */
099: typedef enum {
100:         ACMP_LADDER = 0,
101:         ACMP_I1 = 1,
102:         ACMP_I2 = 2,
103:         ACMP_BANDGAP = 6
104: } acmp_input_t;
105: 
106: /* Hysteresis */
107: typedef enum {
108:         ACMP_NONE,
109:         ACMP_5MV,
110:         ACMP_10MV,
111:         ACMP_20MV
112: } acmp_hys_t;
113: 
114: /* Edge */
115: typedef enum {
116:         ACMP_FALL,
117:         ACMP_RISE,
118:         ACMP_BOTH
119: } acmp_edge_t;
120: 
121: /* Status */
122: enum {
123:         ACMP_COMPSTAT = (1 << 21),
124:         ACMP_COMPEDGE = (1 << 23)
125: };
