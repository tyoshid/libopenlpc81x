032: /* --- PMU registers ------------------------------------------------------- */
033: /*
034:  * Offset       Register
035:  * 0x000        PMU_PCON                Power control register
036:  * 0x004        PMU_GPREG0              General purpose register 0
037:  * 0x008        PMU_GPREG1              General purpose register 1
038:  * 0x00c        PMU_GPREG2              General purpose register 2
039:  * 0x010        PMU_GPREG3              General purpose register 3
040:  * 0x014        PMU_DPDCTRL             Deep power-down control register
041:  */
042: 
043: #define PMU_PCON                        MMIO32(PMU_BASE + 0x000)
044: 
045: #define PMU_GPREG(n)                    MMIO32(PMU_BASE + 0x004 + (n) * 4)
046: #define PMU_GPREG0                      PMU_GPREG(0)
047: #define PMU_GPREG1                      PMU_GPREG(1)
048: #define PMU_GPREG2                      PMU_GPREG(2)
049: #define PMU_GPREG3                      PMU_GPREG(3)
050: 
051: #define PMU_DPDCTRL                     MMIO32(PMU_BASE + 0x014)
052:

055: #define PMU_PCON_DPDFLAG                (1 << 11)
056: #define PMU_PCON_SLEEPFLAG              (1 << 8)

059: #define PMU_PCON_PM1                    (1 << 1)
060: #define PMU_PCON_PM0                    (1 << 0)
