033: /*
034:  * Offset       Register
035:  * 0x000        MRT_INTVAL0             MRT0 Time interval value register
036:  * 0x004        MRT_TIMER0              MRT0 Timer register
037:  * 0x008        MRT_CTRL0               MRT0 Control register
038:  * 0x00c        MRT_STAT0               MRT0 Status register
039:  * 0x010        MRT_INTVAL1             MRT1 Time interval value register
040:  * 0x014        MRT_TIMER1              MRT1 Timer register
041:  * 0x018        MRT_CTRL1               MRT1 Control register
042:  * 0x01c        MRT_STAT1               MRT1 Status register
043:  * 0x020        MRT_INTVAL2             MRT2 Time interval value register
044:  * 0x024        MRT_TIMER2              MRT2 Timer register
045:  * 0x028        MRT_CTRL2               MRT2 Control register
046:  * 0x02c        MRT_STAT2               MRT2 Status register
047:  * 0x030        MRT_INTVAL3             MRT3 Time interval value register
048:  * 0x034        MRT_TIMER3              MRT3 Timer register
049:  * 0x038        MRT_CTRL3               MRT3 Control register
050:  * 0x03c        MRT_STAT3               MRT3 Status register
051:  * 0x0f4        MRT_IDLE_CH             Idle channel register
052:  * 0x0f8        MRT_IRQ_FLAG            Global interrupt flag register
053:  */
