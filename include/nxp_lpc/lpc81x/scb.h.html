023: /* --- System control and ID registers ------------------------------------- */
024: /*
025:  * Offset       Register
026:  * 0xe000e008   SCB_ACTLR               Auxiliary Control Register
027:  *
028:  * 0x000        SCB_CPUID               CPUID Base Register
029:  * 0x004        SCB_ICSR                Interrupt Control State Register
030:  * 0x008        SCB_VTOR                Vector Table Offset Register
031:  * 0x00c        SCB_AIRCR               Application Interrupt and Reset Control
032:  *                                      Register
033:  * 0x010        SCB_SCR                 Optional System Control Register
034:  * 0x014        SCB_CCR                 Configuration and Control Register
035:  * 0x01c        SCB_SHPR2               System Handler Priority Register 2
036:  * 0x020        SCB_SHPR3               System Handler Priority Register 3
037:  * 0x024        SCB_SHCSR               System Handler Control and State
038:  *                                      Register
039:  * 0x030        SCB_DFSR                Debug Fault Status Register
040:  */
041: 
042: #define SCB_ACTLR                       MMIO32(SCS_BASE + 0x008)
043: #define SCB_CPUID                       MMIO32(SCB_BASE + 0x000)
044: #define SCB_ICSR                        MMIO32(SCB_BASE + 0x004)
045: #define SCB_VTOR                        MMIO32(SCB_BASE + 0x008)
046: #define SCB_AIRCR                       MMIO32(SCB_BASE + 0x00c)
047: #define SCB_SCR                         MMIO32(SCB_BASE + 0x010)
048: #define SCB_CCR                         MMIO32(SCB_BASE + 0x014)
049: #define SCB_SHPR2                       MMIO32(SCB_BASE + 0x01c)
050: #define SCB_SHPR3                       MMIO32(SCB_BASE + 0x020)
051: #define SCB_SHCSR                       MMIO32(SCB_BASE + 0x024)
052: #define SCB_DFSR                        MMIO32(SCB_BASE + 0x030)
053:

056: /* --- SCB_CPUID values ---------------------------------------------------- */
057: 
058: /* SCB_CPUID[21:24]: IMPLEMENTER (0x41  ARM Limited.) */
059: /* SCB_CPUID[23:20]: VARIANT */
060: /* SCB_CPUID[19:16]: ARCHITECTURE (0xc  ARMv6-M) */
061: /* SCB_CPUID[15:4]: PARTNO */
062: /* SCB_CPUID[3:0]: REVISION */

071: #define SCB_ICSR_ISRPREEMPT             (1 << 23)
072: #define SCB_ICSR_ISRPENDING             (1 << 22)
073: /* SCB_ICSR[20:12]: VECTPENDING */
074: /* SCB_ICSR[8:0]: VECTACTIVE */

077: 
078: /* SCB_VTOR[31:7]: TBLOFF */
