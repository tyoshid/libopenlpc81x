033: /*
034:  * Offset       Register
035:  * 0x000        I2C_CFG                 Configuration register
036:  * 0x004        I2C_STAT                Status register
037:  * 0x008        I2C_INTENSET            Interrupt Enable Set and read register
038:  * 0x00c        I2C_INTENCLR            Interrupt Enable Clear register
039:  * 0x010        I2C_TIMEOUT             Time-out value register
040:  * 0x014        I2C_CLKDIV              Clock Divider register
041:  * 0x018        I2C_INTSTAT             Interrupt Status register
042:  * 0x020        I2C_MSTCTL              Master Control register
043:  * 0x024        I2C_MSTTIME             Master Time register
044:  * 0x028        I2C_MSTDAT              Master Data register
045:  * 0x040        I2C_SLVCTL              Slave Control register
046:  * 0x044        I2C_SLVDAT              Slave Data register
047:  * 0x048        I2C_SLVADR0             Slave Address register 0
048:  * 0x04c        I2C_SLVADR1             Slave Address register 1
049:  * 0x050        I2C_SLVADR2             Slave Address register 2
050:  * 0x054        I2C_SLVADR3             Slave Address register 3
051:  * 0x058        I2C_SLVQUAL0            Slave address Qualifier 0 register
052:  * 0x080        I2C_MONRXDAT            Monitor receiver data register
053:  */
