032: /* --- WWDT registers ------------------------------------------------------ */
033: /*
034:  * Offset       Register
035:  * 0x000        WWDT_MOD                Watchdog mode register
036:  * 0x004        WWDT_TC                 Watchdog timer constat register
037:  * 0x008        WWDT_FEED               Watchdog feed sequence register
038:  * 0x00c        WWDT_TV                 Watchdog timer value register
039:  * 0x014        WWDT_WARNINT            Watchdog Warning Interrupt register
040:  * 0x018        WWDT_WINDOW             Watchdog Window register
041:  */
042: 
043: #define WWDT_MOD                        MMIO32(WWDT_BASE + 0x000)
044: #define WWDT_TC                         MMIO32(WWDT_BASE + 0x004)
045: #define WWDT_FEED                       MMIO32(WWDT_BASE + 0x008)
046: #define WWDT_TV                         MMIO32(WWDT_BASE + 0x00c)
047: #define WWDT_WARNINT                    MMIO32(WWDT_BASE + 0x014)
048: #define WWDT_WINDOW                     MMIO32(WWDT_BASE + 0x018)
049: 
050: /* --- WWDT_MOD values ----------------------------------------------------- */
051: 
052: #define WWDT_MOD_LOCK                   (1 << 5)
053: #define WWDT_MOD_WDPROTECT              (1 << 4)
054: #define WWDT_MOD_WDINT                  (1 << 3)
055: #define WWDT_MOD_WDTOF                  (1 << 2)
056: #define WWDT_MOD_WDRESET                (1 << 1)
057: #define WWDT_MOD_WDEN                   (1 << 0)
058: 
059: /* --- WWDT_TC values ------------------------------------------------------ */
060: 
061: /* WWDT_TC[23:0]: COUNT */
062: 
063: /* --- WWDT_FEED values --------------------------------------------------- */
064: 
065: /* WWDT_FEED[7:0]: FEED (0xaa, 0x55) */
066:

068: 
069: /* WWDT_TV[23:0]: COUNT */
070:

072: 
073: /* WWDT_WARNINT[9:0]: WARNINT */
074:
