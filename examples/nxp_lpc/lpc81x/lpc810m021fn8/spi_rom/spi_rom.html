026: /*
027:  * Microchip 25LC640A
028:  * 64Kbit SPI Serial EEPROM
029:  *
030:  * Vcc = 3.3V
031:  * FCLKmax      5MHz            Clock frequency
032:  * TCSSmin      100nsec         CS setup time
033:  * TCSHmin      200nsec         CS hold time
034:  * TCSDmin       50nsec         CS disable time
035:  */
036: #define ROM_SIZE        8192    /* 8KByte */
037: #define PAGE_SIZE       32      /* 32Byte */
038: #define COMMAND_READ    0x03
039: #define COMMAND_WRITE   0x02
040: #define COMMAND_WRDI    0x04
041: #define COMMAND_WREN    0x06
042: #define COMMAND_RDSR    0x05
043: #define COMMAND_WRSR    0x01
044: #define TWC             5       /* Internal write cycle time: 5 msec */
045: 
046: /* USART clock frequency */
047: #define U_PCLK  (16 * 115200)
048: 
049: /* Set LPC810 to 24 MHz. */
050: static void clock_setup(void)
051: {
052:         /* Set up PLL (Fclkin = 12MHz (IRC), Fclkout (Main clock) = 24MHz) */
053:         syscon_enable_pll(SYSCON_IRC, 2, 4);
054: 
055:         /* Select PLL as system clock source. */
056:         syscon_set_system_clock(SYSCON_PLL_OUT, 1);
057: 
058:         /* Enable USART clock (U_PCLK). */
059:         syscon_set_usart_clock(24000000, U_PCLK);
060: }
061: 
062: static void gpio_setup(void)
063: {
064:         /* Enable GPIO clock. */
065:         // syscon_enable_clock(SYSCON_GPIO);
066: 
067:         /* Enable switch matrix clock. */
068:         // syscon_enable_clock(SYSCON_SWM);
069: 
070:         /* Enable IOCON clock. */
071:         syscon_enable_clock(SYSCON_IOCON);
072: 
073:         /* Prevent the pins from internally floating. */
074:         gpio_config(GPIO_OUTPUT, GPIO_IO, PIO0_10 | PIO0_11);
075:         gpio_clear(PIO0_10 | PIO0_11);
076: 
077:         /* Set PIO0_2 to 'SPI0 SCK'. */
078:         gpio_config(GPIO_SPI0_SCK, GPIO_HYST, PIO0_2);
079: 
080:         /* Set PIO0_3 to 'SPIO MOSI'. */
081:         gpio_config(GPIO_SPI0_MOSI, GPIO_HYST, PIO0_3);
082: 
083:         /* Set PIO0_5 to 'SPIO MISO'. */
084:         gpio_config(GPIO_SPI0_MISO, GPIO_HYST, PIO0_5);
085: 
086:         /* Set PIO0_0 to 'SPIO SSEL'. */
087:         gpio_config(GPIO_SPI0_SSEL, GPIO_HYST, PIO0_0);
088: 
089:         /* Set PIO0_4 to 'USART0 TXD'. */
090:         gpio_config(GPIO_U0_TXD, GPIO_HYST, PIO0_4);
091: }
092: 
093: static void usart_setup(void)
094: {
095:         /* Enable USART0 clock. */
096:         syscon_enable_clock(SYSCON_UART0);
097: 
098:         /* Set up USART0. */
099:         usart_init(USART0, U_PCLK, 115200, 8, 1, USART_PARITY_NONE,
100:                    USART_FLOW_NONE);
101: }

103: static void mrt_setup(void)
104: {
105:         /* Enable MRT clock. */
106:         syscon_enable_clock(SYSCON_MRT);
107: 
108:         /* Set channel 0 to 'one-shot interrupt mode'. */
109:         mrt_set_mode(MRT0, MRT_ONE_SHOT);
110: }

112: static void spi_setup(void)
113: {
114:         /* Enable SPI0 clock. */
115:         syscon_enable_clock(SYSCON_SPI0);
116: 
117:         /* Enable SPI0. */
118:         /* 24MHz / 6 = 4MHz */
119:         spi_init_master(SPI0, SPI_MODE0, 6, 0, 0, 0, 0);
120: }

122: static void print_hex(int c)
123: {
124:         int d;
125: 
126:         d = (c >> 4) & 0xf;
127:         if (d > 9)
128:                 d += 'a' - 10;
129:         else
130:                 d += '0';
131:         usart_send_blocking(USART0, d);
132:         d = c & 0xf;
133:         if (d > 9)
134:                 d += 'a' - 10;
135:         else
136:                 d += '0';
137:         usart_send_blocking(USART0, d);
138: }
139: 
140: static void print_hex_half_word(int w)
141: {
142:         print_hex((w >> 8) & 0xff);
143:         print_hex(w & 0xff);
144: }
145: 
146: static void print_hex_word(int w)
147: {
148:         print_hex((w >> 24) & 0xff);
149:         print_hex((w >> 16) & 0xff);
150:         print_hex((w >> 8) & 0xff);
151:         print_hex(w & 0xff);
152: }

154: static void print_int(int d)
155: {
156:         bool zerosup;
157:         int i;
158: 
159:         if (d < 0) {
160:                 usart_send_blocking(USART0, '-');
161:                 d = -d;
162:         }
163:         zerosup = true;
164:         for (i = 1000000000; i > 0; i /= 10) {
165:                 if (d / i || i == 1 || !zerosup) {
166:                         usart_send_blocking(USART0, d / i + '0');
167:                         d %= i;
168:                         zerosup = false;
169:                 }
170:         }
171: }

173: static void xprintf(char *p, int arg0, int arg1, int arg2)
174: {
175:         int arg[3];
176:         int i;
177: 
178:         arg[0] = arg0;
179:         arg[1] = arg1;
180:         arg[2] = arg2;
181:         i = 0;
182:         while (*p != '\0') {
183:                 if (*p == '%') {
184:                         switch (*++p) {
185:                         case '%':
186:                                 usart_send_blocking(USART0, *p++);
187:                                 break;
188:                         case 'x':
189:                                 if (i < 3)
190:                                         print_hex(arg[i++]);
191:                                 p++;
192:                                 break;
193:                         case 'h':
194:                                 if (i < 3)
195:                                         print_hex_half_word(arg[i++]);
196:                                 p++;
197:                                 break;
198:                         case 'w':
199:                                 if (i < 3)
200:                                         print_hex_word(arg[i++]);
201:                                 p++;
202:                                 break;
203:                         case 'd':
204:                                 if (i < 3)
205:                                         print_int(arg[i++]);
206:                                 p++;
207:                                 break;
208:                         default:
209:                                 usart_send_blocking(USART0, '?');
210:                                 p++;
211:                                 break;
212:                         }
213:                 } else {
214:                         usart_send_blocking(USART0, *p++);
215:                 }
216:         }
217: }

219: static void xputs(char *p)
220: {
221:         while (*p != '\0')
222:                 usart_send_blocking(USART0, *p++);
223:         usart_send_blocking(USART0, '\r');
224:         usart_send_blocking(USART0, '\n');
225: }

227: static void delay_ms(int ms)
228: {
229:         mrt_set_interval(MRT0, 24000000 / 1000 * ms);
230: 
231:         while (!mrt_get_channel_status(MRT0, MRT_INT))
232:                 ;
233: 
234:         mrt_clear_channel_status(MRT0, MRT_INT);
235: }

237: /* Read */
238: static void rom_read(int addr, char *data, int nbyte)
239: {
240:         int i;
241: 
242:         /* Instruction */
243:         spi_send_control_blocking(SPI0, COMMAND_READ, 8, SPI_RXIGNORE);
244: 
245:         /* Address */
246:         spi_send_control_blocking(SPI0, addr, 16, SPI_RXIGNORE);
247: 
248:         /* Data */
249:         spi_set_tx_control(SPI0, 8, 0);
250:         for (i = 0; i < nbyte; i++) {
251:                 if (i == nbyte - 1)
252:                         spi_set_tx_control(SPI0, 8, SPI_EOT);
253:                 spi_send_blocking(SPI0, 0);
254:                 *data++ = spi_recv_blocking(SPI0);
255:         }
256: }
257: 
258: /* Byte Write */
259: static void rom_write_byte(int addr, int data)
260: {
261:         /* Write Enable */
262:         spi_send_control_blocking(SPI0, COMMAND_WREN, 8,
263:                                   SPI_RXIGNORE | SPI_EOT);
264: 
265:         /* Instruction */
266:         spi_send_control_blocking(SPI0, COMMAND_WRITE, 8, SPI_RXIGNORE);
267: 
268:         /* Address */
269:         spi_send_control_blocking(SPI0, addr, 16, SPI_RXIGNORE);
270: 
271:         /* Data */
272:         spi_send_control_blocking(SPI0, data, 8, SPI_RXIGNORE | SPI_EOT);
273: }
274: 
275: /* Page Write */
276: static void rom_write_page(int addr, char *data)
277: {
278:         int i;
279: 
280:         /* Write Enable */
281:         spi_send_control_blocking(SPI0, COMMAND_WREN, 8,
282:                                   SPI_RXIGNORE | SPI_EOT);
283: 
284:         /* Instruction */
285:         spi_send_control_blocking(SPI0, COMMAND_WRITE, 8, SPI_RXIGNORE);
286: 
287:         /* Address */
288:         spi_send_control_blocking(SPI0, addr, 16, SPI_RXIGNORE);
289: 
290:         /* Data */
291:         spi_set_tx_control(SPI0, 8, SPI_RXIGNORE);
292:         for (i = 0; i < PAGE_SIZE; i++) {
293:                 if (i == PAGE_SIZE - 1)
294:                         spi_set_tx_control(SPI0, 8, SPI_RXIGNORE | SPI_EOT);
295:                 spi_send_blocking(SPI0, *data++);
296:         }
297: }

299: int main(void)
300: {
301:         char start_data;
302:         int addr;
303:         char data;
304:         int i;
305:         int j;
306:         char buf[256];
307:         int n;
308:         int m;
309: 
310:         clock_setup();
311:         gpio_setup();
312:         usart_setup();
313:         mrt_setup();
314:         spi_setup();
315: 
316:         xputs("spi_rom");
317: 
318:         xputs("Read 1 byte");
319:         rom_read(0, &start_data, 1);
320:         start_data++;
321:         xprintf("data = 0x%x\r\n", start_data, 0, 0);
322: 
323:         /* Write data. */
324:         addr = 0;
325:         data = start_data;
326:         /* Byte */
327:         xputs("Byte Write");
328:         for (i = 0; i < PAGE_SIZE; i++) {
329:                 rom_write_byte(addr++, data++);
330: 
331:                 /* internal write cycle time */
332:                 delay_ms(TWC);
333:         }
334: 
335:         /* Page */
336:         xputs("Page Write");
337:         for (i = 0; i < ROM_SIZE / PAGE_SIZE - 1; i++) {
338:                 for (j = 0; j < PAGE_SIZE; j++)
339:                         buf[j] = data++;
340:                 rom_write_page(addr, buf);
341:                 addr += PAGE_SIZE;
342: 
343:                 /* internal write cycle time */
344:                 delay_ms(TWC);
345:         }
346: 
347:         /* Read and verify data. */
348:         xputs("Sequential Read");
349:         while (1) {
350:                 for (n = 1; n < (int)sizeof(buf); n++) {
351:                         xprintf("size = %d\r\n", n, 0, 0);
352: 
353:                         addr = 0;
354:                         data = start_data;
355:                         for (i = 0; i < ROM_SIZE / n; i++) {
356:                                 rom_read(addr, buf, n);
357:                                 for (j = 0; j < n; j++) {
358:                                         if (buf[j] != data)
359:                                                 goto data_error;
360:                                         data++;
361:                                 }
362:                                 addr += n;
363:                         }
364: 
365:                         m = ROM_SIZE % n;
366:                         if (m) {
367:                                 rom_read(addr, buf, m);
368:                                 for (j = 0; j < m; j++) {
369:                                         if (buf[j] != data)
370:                                                 goto data_error;
371:                                         data++;
372:                                 }
373:                         }
374:                 }
375:         }
376: 
377: data_error:
378:         xprintf("data error: addr = 0x%h  data = 0x%x (0x%x)\r\n",
379:                 addr + j, buf[j], data);
380:         while (1)
381:                 ;
382: 
383:         return 0;
384: }
