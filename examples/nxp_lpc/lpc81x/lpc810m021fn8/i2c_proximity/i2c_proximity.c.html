057: static void clock_setup(void)
058: {
059:         /* Set up PLL (Fclkin = 12MHz (IRC), Fclkout (Main clock) = 24MHz) */
060:         syscon_enable_pll(SYSCON_IRC, 2, 4);
061: 
062:         /* Select PLL as system clock source. */

065:         /* Enable USART clock (U_PCLK). */
066:         syscon_set_usart_clock(24000000, U_PCLK);

069:         /* Set IOCON glitch filter clock divider. */
070:         syscon_set_ioconclkdiv(0, 1);

074: static void gpio_setup(void)
075: {
076:         /* Enable GPIO clock. */
077:         // syscon_enable_clock(SYSCON_GPIO);
078: 
079:         /* Enable switch matrix clock. */
080:         // syscon_enable_clock(SYSCON_SWM);
081: 
082:         /* Enable IOCON clock. */
083:         syscon_enable_clock(SYSCON_IOCON);
084: 
085:         /* Prevent the pins from internally floating. */
086:         gpio_config(GPIO_OUTPUT, GPIO_IO, PIO0_10 | PIO0_11);
087:         gpio_clear(PIO0_10 | PIO0_11);
088: 
089: #ifndef FASTMODE
090:         /* Set PIO0_2 to 'I2C SCL'. */
091:         gpio_config(GPIO_I2C_SCL, GPIO_OPENDRAIN | GPIO_HYST, PIO0_2);
092: 
093:         /* Set PIO0_3 to 'I2C SDA'. */
094:         gpio_config(GPIO_I2C_SDA, GPIO_OPENDRAIN | GPIO_HYST, PIO0_3);
095: #else
096:         /* Set PIO0_2 to 'I2C SCL'. */
097:         gpio_config(GPIO_I2C_SCL, GPIO_OPENDRAIN | GPIO_HYST |
098:                     GPIO_IOCONCLKDIV0 | GPIO_FILTER_2CLOCK, PIO0_2);
099: 
100:         /* Set PIO0_3 to 'I2C SDA'. */
101:         gpio_config(GPIO_I2C_SDA, GPIO_OPENDRAIN | GPIO_HYST |
102:                     GPIO_IOCONCLKDIV0 | GPIO_FILTER_2CLOCK, PIO0_3);
103: #endif
104: 
105:         /* Set PIO0_4 to 'USART0 TXD'. */
106:         gpio_config(GPIO_U0_TXD, GPIO_HYST, PIO0_4);
107: }

153: static void print_hex(int c)
154: {
155:         int d;
156: 
157:         d = (c >> 4) & 0xf;
158:         if (d > 9)
159:                 d += 'a' - 10;
160:         else
161:                 d += '0';
162:         usart_send_blocking(USART0, d);
163:         d = c & 0xf;
164:         if (d > 9)
165:                 d += 'a' - 10;
166:         else
167:                 d += '0';
168:         usart_send_blocking(USART0, d);
169: }

185: static void print_int(int d)
186: {
187:         bool zerosup;
188:         int i;
189: 
190:         if (d < 0) {
191:                 usart_send_blocking(USART0, '-');
192:                 d = -d;
193:         }
194:         zerosup = true;
195:         for (i = 1000000000; i > 0; i /= 10) {
196:                 if (d / i || i == 1 || !zerosup) {
197:                         usart_send_blocking(USART0, d / i + '0');
198:                         d %= i;
199:                         zerosup = false;
200:                 }
201:         }
202: }

204: static void xprintf(char *p, int arg0, int arg1, int arg2)
205: {
206:         int arg[3];
207:         int i;
208: 
209:         arg[0] = arg0;
210:         arg[1] = arg1;
211:         arg[2] = arg2;
212:         i = 0;
213:         while (*p != '\0') {
214:                 if (*p == '%') {
215:                         switch (*++p) {
216:                         case '%':
217:                                 usart_send_blocking(USART0, *p++);
218:                                 break;
219:                         case 'x':
220:                                 if (i < 3)
221:                                         print_hex(arg[i++]);
222:                                 p++;
223:                                 break;
224:                         case 'h':
225:                                 if (i < 3)
226:                                         print_hex_half_word(arg[i++]);
227:                                 p++;
228:                                 break;
229:                         case 'w':
230:                                 if (i < 3)
231:                                         print_hex_word(arg[i++]);
232:                                 p++;
233:                                 break;
234:                         case 'd':
235:                                 if (i < 3)
236:                                         print_int(arg[i++]);
237:                                 p++;
238:                                 break;
239:                         default:
240:                                 usart_send_blocking(USART0, '?');
241:                                 p++;
242:                                 break;
243:                         }
244:                 } else {
245:                         usart_send_blocking(USART0, *p++);
246:                 }
247:         }
248: }

273: void i2c_isr(void)
274: {
275:         int m;
276:         int s;
277: 
278:         /* Get interrupt mask. */
279:         m = i2c_get_interrupt_mask(I2C_MSTSTSTPERR | I2C_MSTARBLOSS |
280:                                    I2C_MSTPENDING);
281: 
282:         /* Get interrupt status. */
283:         s = i2c_get_interrupt_status(I2C_MSTSTSTPERR | I2C_MSTARBLOSS |
284:                                      I2C_MSTPENDING);
285: 
286:         /* Master Pending */
287:         if (m & s & I2C_MSTPENDING) {
288:                 /* Get Master State code. */
289:                 state = i2c_get_status(I2C_MSTSTATE);
290: 
291:                 /* unexpected state */
292:                 if (state != next_state) {
293:                         /* Disable interrupt. */
294:                         i2c_disable_interrupt(I2C_MSTPENDING);
295:                                 
296:                         /* Clear flag. */
297:                         busy = false;
298:                 }
299: 
300:                 switch (state) {
301:                 case I2C_MST_IDLE:
302:                         if (sndbuflen) {
303:                                 /* Send START + I2C Slave address (W). */
304:                                 i2c_master_send_addr(VCNL4000_SLA);
305: 
306:                                 next_state = I2C_MST_TX_RDY;
307:                         } else if (rcvbuflen) {
308:                                 /* Send START + I2C Slave address (R). */
309:                                 i2c_master_send_addr(VCNL4000_SLA | 1);
310: 
311:                                 next_state = I2C_MST_RX_RDY;
312:                         } else {
313:                                 /* transfer end */
314: 
315:                                 /* Disable interrupt. */
316:                                 i2c_disable_interrupt(I2C_MSTPENDING);
317:                                 
318:                                 /* Clear flag. */
319:                                 busy = false;
320:                         }
321:                         break;
322:                 case I2C_MST_RX_RDY:
323:                         if (rcvbuflen > 1) {
324:                                 /* Get data and send ACK. */
325:                                 *rcvbuf++ = i2c_master_recv();
326:                                 rcvbuflen--;
327: 
328:                                 next_state = I2C_MST_RX_RDY;
329:                         } else {
330:                                 /* Get data. */
331:                                 *rcvbuf = i2c_master_recv_last();
332:                                 rcvbuflen--;
333: 
334:                                 /* Send NACK + STOP. */
335:                                 i2c_stop();
336: 
337:                                 next_state = I2C_MST_IDLE;
338:                         }
339:                 case I2C_MST_TX_RDY:
340:                         if (sndbuflen) {
341:                                 /* Send data. */
342:                                 i2c_master_send(*sndbuf++);
343:                                 sndbuflen--;
344: 
345:                                 next_state = I2C_MST_TX_RDY;
346:                         } else {
347:                                 /* Send STOP. */
348:                                 i2c_stop();
349: 
350:                                 next_state = I2C_MST_IDLE;
351:                         }
352:                         break;
353:                 default:
354:                         break;
355:                 }
356:         }
357: 
358:         /* Master Start/Stop Error */
359:         if (m & s & I2C_MSTSTSTPERR) {
360:                 start_stop_error++;
361:                 i2c_clear_status(I2C_MSTSTSTPERR);
362:         }
363: 
364:         /* Master Arbitration Loss */
365:         if (m & s & I2C_MSTARBLOSS) {
366:                 arbitration_loss++;
367:                 i2c_clear_status(I2C_MSTARBLOSS);
368:         }
369: }

371: /* Write command */
372: static int vcnl_write_command(int addr, int data)
373: {
374:         int buf[2];
375: 
376:         /* Write register address and data. */
377:         buf[0] = addr;
378:         buf[1] = data;
379:         sndbuf = buf;
380:         sndbuflen = 2;
381: 
382:         /* nothing to read */
383:         rcvbuflen = 0;
384: 
385:         /* Set flag. */
386:         busy = true;
387: 
388:         /* Idle */
389:         next_state = I2C_MST_IDLE;
390: 
391:         /* Enable interrupt. */
392:         i2c_enable_interrupt(I2C_MSTPENDING);
393: 
394:         /* Wait until the completion of the transfer. */
395:         while (busy)
396:                 ;
397: 
398:         /* Error */
399:         if (state != next_state)
400:                 return -1;
401: 
402:         return 0;
403: }

405: /* Read word */
406: static int vcnl_read_word(int addr, int *data)
407: {
408:         int h;
409: 
410:         /* Write register address. */
411:         sndbuf = &addr;
412:         sndbuflen = 1;
413: 
414:         /* Read high byte. */
415:         rcvbuf = &h;
416:         rcvbuflen = 1;
417: 
418:         /* Set flag. */
419:         busy = true;
420: 
421:         /* Idle */
422:         next_state = I2C_MST_IDLE;
423: 
424:         /* Enable interrupt. */
425:         i2c_enable_interrupt(I2C_MSTPENDING);
426: 
427:         /* Wait until the completion of the transfer. */
428:         while (busy)
429:                 ;
430: 
431:         /* Error */
432:         if (state != next_state)
433:                 return -1;
434: 
435:         /* nothing to write */
436:         sndbuflen = 0;
437: 
438:         /* Read low byte. */
439:         rcvbuf = data;
440:         rcvbuflen = 1;
441: 
442:         /* Set flag. */
443:         busy = true;
444: 
445:         /* Enable interrupt. */
446:         i2c_enable_interrupt(I2C_MSTPENDING);
447: 
448:         /* Wait until the completion of the transfer. */
449:         while (busy)
450:                 ;
451: 
452:         /* Error */
453:         if (state != next_state)
454:                 return -1;
455: 
456:         /* 16bit data */
457:         *data |= h << 8;
458: 
459:         return 0;
460: }

462: int main(void)
463: {
464:         int d;
465:         int i;
466:         int prox_min;
467:         int prox_max;
468:         int total;
469:         int prox_off;
470:         int n;
471:         int x;
472:         int p;
473: 
474:         clock_setup();
475:         gpio_setup();
476:         usart_setup();
477:         mrt_setup();
478:         i2c_setup();
479: 
480:         xputs("i2c_proximity");
481: 
482:         /* Power Up */
483:         delay_us(400);
484: 
485:         /* Initialize */
486:         /* Set IR LED current to 100mA. */
487:         if (vcnl_write_command(0x83, 10)) {
488:                 xputs("register write error");
489:                 goto i2c_error;
490:         }
491:         /* Set proximity IR test signal frequency to 781.25 kHz. */
492:         if (vcnl_write_command(0x89, 2)) {
493:                 xputs("register write error");
494:                 goto i2c_error;
495:         }
496:         /* Modulation delay time = 4 and Modulation dead time = 1 */
497:         if (vcnl_write_command(0x8a, 129)) {
498:                 xputs("register write error");
499:                 goto i2c_error;
500:         }
501: 
502:         /* Proximity offset */
503:         for (i = 0; i < 5; i++) {
504:                 /* Start measurement. */
505:                 if (vcnl_write_command(0x80, 8)) {
506:                         xputs("register write error");
507:                         goto i2c_error;
508:                 }
509: 
510:                 delay_us(i == 0 ? 600 : 200);
511: 
512:                 /* Get measurement result */
513:                 if (vcnl_read_word(0x87, &d)) {
514:                         xputs("register read error");
515:                         goto i2c_error;
516:                 }
517:                 if (i == 0) {
518:                         prox_min = d;
519:                         prox_max = d;
520:                         total = d;
521:                 } else {
522:                         if (d < prox_min)
523:                                 prox_min = d;
524:                         if (d > prox_max)
525:                                 prox_max = d;
526:                         total += d;
527:                 }
528: 
529:                 delay_ms(10);
530:         }
531:         prox_off = total / 5;
532:         xprintf("proximity offset = %d (%d - %d)\r\n",
533:                 prox_off, prox_min, prox_max);
534: 
535:         /* Ambient Light */
536:         /* Start measurement. */
537:         if (vcnl_write_command(0x80, 0x10)) {
538:                 xputs("register write error");
539:                 goto i2c_error;
540:         }
541: 
542:         delay_ms(400);
543: 
544:         /* Get measurement result */
545:         if (vcnl_read_word(0x85, &d)) {
546:                 xputs("register read error");
547:                 goto i2c_error;
548:         }
549:         xprintf("Ambient Light = %d lx\r\n", d / 4, 0, 0);
550: 
551:         while (1) {
552:                 n = 0;
553:                 x = 0;
554:                 p = 0;
555:                 while (n < 9 && x < 10) {
556:                         /* Start measurement. */
557:                         if (vcnl_write_command(0x80, 8)) {
558:                                 xputs("register write error");
559:                                 goto i2c_error;
560:                         }
561: 
562:                         delay_us(200);
563: 
564:                         /* Get measurement result */
565:                         if (vcnl_read_word(0x87, &d)) {
566:                                 xputs("register read error");
567:                                 goto i2c_error;
568:                         }
569: 
570:                         if (d > prox_off + PROX_THRESH) {
571:                                 p += d;
572:                                 if (++x == 10) {
573:                                         xprintf("Proximity detected %d\r\n",
574:                                                 p / 10, 0, 0);
575:                                 }
576:                         } else {
577:                                 n++;
578:                         }
579: 
580:                         delay_ms(10);
581:                 }
582:         }
583: 
584: i2c_error:
585:         xprintf("I2C error: state = 0x%x  expected state = 0x%x\r\n",
586:                 state, next_state, 0);
587:         while (1)
588:                 ;
589: 
590:         return 0;
591: }
