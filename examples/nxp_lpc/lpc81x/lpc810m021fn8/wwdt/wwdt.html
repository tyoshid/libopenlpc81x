034: /* Set LPC810 to 24 MHz (+/- 1.5%). */
035: static void clock_setup(void)
036: {
037:         /* Set up PLL (Fclkin = 12MHz (IRC), Fclkout (Main clock) = 24MHz) */
038:         syscon_enable_pll(SYSCON_IRC, 2, 4);
039: 
040:         /* Select PLL as system clock source. */
041:         syscon_set_system_clock(SYSCON_PLL_OUT, 1);
042: 
043:         /* wdt_osc_clk = Fclkana / (2 * (1 + DIVSEL)) */
044:         if (TESTNUM == 4 - 1)
045:                 /* 0.6MHz / (2 * (1 + 31)) = 9.3kHz (+/- 40%) */
046:                 syscon_enable_wdt_osc(SYSCON_0_6MHZ, 31);
047:         else
048:                 /* 4.6MHz / (2 * (1 + 0)) = 2.3MHz (+/- 40%) */
049:                 syscon_enable_wdt_osc(SYSCON_4_6MHZ, 0);
050: }

052: static void gpio_setup(void)
053: {
054:         /* Enable GPIO clock. */
055:         // syscon_enable_clock(SYSCON_GPIO);
056: 
057:         /* Enable switch matrix clock. */
058:         // syscon_enable_clock(SYSCON_SWM);
059: 
060:         /* Enable IOCON clock. */
061:         syscon_enable_clock(SYSCON_IOCON);
062: 
063:         /* Prevent the I2C pins from internally floating. */
064:         gpio_config(GPIO_OUTPUT, GPIO_IO, PIO0_10 | PIO0_11);
065:         gpio_clear(PIO0_10 | PIO0_11);
066: 
067:         /* Set PIO0_2 and PIO0_3 to 'output push-pull' (LED). */
068:         gpio_config(GPIO_OUTPUT, 0, PIO0(2, 3));
069: }

433: int main(void)
434: {
435:         int s;
436: 
437:         /* Get system reset status. */
438:         s = syscon_get_reset_status(SYSCON_WDT_RST | SYSCON_SYSRST);
439: 
440:         /* Clear system reset status. */
441:         syscon_clear_reset_status(SYSCON_WDT_RST | SYSCON_SYSRST);
442:         
443:         clock_setup();
444:         gpio_setup();
445:         mrt_setup();
446:         wwdt_setup();
447: 
448:         if (s & SYSCON_WDT_RST) {
449:                 /* watchdog reset */
450:                 if (STATE == 0)
451:                         goto wwdt_error0;
452:                 if (STATE == 2)
453:                         goto wwdt_error2;
454:                 if (TESTNUM == 5)
455:                         goto wwdt_error0;
456: 
457:                 /* next test */
458:                 if (TESTNUM >= 5)
459:                         TESTNUM = 0;
460:                 else
461:                         TESTNUM++;
462:         } else if (s & SYSCON_SYSRST) {
463:                 /* software system reset */
464:                 if (TESTNUM != 5)
465:                         goto wwdt_error0;
466:                 if (STATE == 0)
467:                         goto wwdt_error0;
468:                 if (STATE == 2)
469:                         goto wwdt_error2;
470:                 if (INTSTATUS != (WWDT_TIME_OUT | WWDT_WARNING))
471:                         goto wwdt_error2;
472: 
473:                 INTSTATUS = 0;
474:                 TESTNUM = 0;
475:         } else {
476:                 /* power-on reset */
477:                 TESTNUM = 0;
478:         }
479: 
480:         switch (TESTNUM) {
481:         case 0:
482:                 test0();
483:                 break;
484:         case 1:
485:                 test1();
486:                 break;
487:         case 2:
488:                 test2();
489:                 break;
490:         case 3:
491:                 test3();
492:                 break;
493:         case 4:
494:                 test4();
495:                 break;
496:         case 5:
497:                 test5();
498:                 break;
499:         default:
500:                 break;
501:         }
502: 
503: wwdt_error2:
504:         while (1) {
505:                 /* Blink LED2 slowly. */
506:                 gpio_toggle(PIO0_3);
507:                 delay_ms(2000);
508:         }
509: 
510: wwdt_error0:
511:         while (1) {
512:                 /* Blink LED2 quickly. */
513:                 gpio_toggle(PIO0_3);
514:                 delay_ms(500);
515:         }
516: 
517:         return 0;
518: }
