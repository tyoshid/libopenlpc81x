020: #include <syscon.h>
021: #include <gpio.h>
022: #include <nvic.h>
023: #include <pinint.h>
024: 
025: /* Set LPC810 to 24 MHz. */
026: static void clock_setup(void)
027: {
028:         /* Set up PLL (Fclkin = 12MHz (IRC), Fclkout (Main clock) = 24MHz) */
029:         syscon_enable_pll(SYSCON_IRC, 2, 4);
030: 
031:         /* Select PLL as system clock source. */
032:         syscon_set_system_clock(SYSCON_PLL_OUT, 1);
033: }
034: 
035: static void gpio_setup(void)
036: {
037:         /* Enable GPIO clock. */
038:         // syscon_enable_clock(SYSCON_GPIO);
039: 
040:         /* Enable switch matrix clock. */
041:         // syscon_enable_clock(SYSCON_SWM);
042: 
043:         /* Enable IOCON clock. */
044:         syscon_enable_clock(SYSCON_IOCON);
045: 
046:         /* Prevent the I2C pins from internally floating. */
047:         gpio_config(GPIO_OUTPUT, GPIO_IO, PIO0_10 | PIO0_11);
048:         gpio_clear(PIO0_10 | PIO0_11);
049: 
050:         /* Set PIO0_2 to 'output push-pull' (LED). */
051:         gpio_config(GPIO_OUTPUT, 0, PIO0_2);
052: 
053:         /* Set PIO0_3 to 'input float' (SW). */
054:         gpio_config(GPIO_INPUT, GPIO_HYST, PIO0_3);
055: }
056: 
057: static void pinint_setup(void)
058: {
059:         /* Set PININT0 to PIO0_3. */
060:         syscon_select_pins(0, PIO0_3);
061: 
062:         /* Enable interrupt. */
063:         /* PININT0: Pin interrupt */
064:         /* PININT1: Pattern match */
065:         nvic_enable_irq(NVIC_PININT0 | NVIC_PININT1);
066: 
067:         /* Clear interrupt */
068:         pinint_clear_interrupt(PININT0);
069: 
070:         /* Enable falling edge interrupt. */
071:         pinint_set_interrupt_mode(PININT_FALLING, PININT0);
072: }
073: 
074: /*
075:  * Change the interrupt mode when the button is pushed and released 3 times.
076:  * 1) falling edge (pin interrupt)
077:  * 2) rising edge (pin interrupt)
078:  * 3) both edge (pin interrupt)
079:  * 4) falling edge (pattern match)
080:  * 5) rising edge (pattern match)
081:  * 6) both edge (pattern match)
082:  */
083: 
084: /* Pin interrupt */
085: void pinint0_isr(void)
086: {
087:         pinint_mode_t m;
088:         static int count;
089: 
090:         /* Not pattern match! */
091:         if (pinint_pattern_match_enabled())
092:                 return;
093: 
094:         m = pinint_get_interrupt_mode(PININT0);
095:         if (m == PININT_DISABLE || !pinint_get_interrupt_status(PININT0))
096:                 return;
097: 
098:         switch (m) {
099:         case PININT_FALLING:
100:                 if (!pinint_falling_edge_detected(PININT0))
101:                         break;
102: 
103:                 /* Blink the LED (PIO0_2). */
104:                 gpio_toggle(PIO0_2);
105: 
106:                 if (++count >= 3) {
107:                         count = -1;
108: 
109:                         /* Enable rising edge interrupt. */
110:                         pinint_set_interrupt_mode(PININT_RISING, PININT0);
111:                 }
112:                 break;
113:         case PININT_RISING:
114:                 if (!pinint_rising_edge_detected(PININT0))
115:                         break;
116: 
117:                 if (count >= 0)
118:                         /* Blink the LED (PIO0_2). */
119:                         gpio_toggle(PIO0_2);
120: 
121:                 if (++count >= 3) {
122:                         count = 0;
123: 
124:                         /* Enable both edge interrupt. */
125:                         pinint_set_interrupt_mode(PININT_BOTH, PININT0);
126:                 }
127:                 break;
128:         case PININT_BOTH:
129:                 if (!pinint_rising_edge_detected(PININT0) &&
130:                     !pinint_falling_edge_detected(PININT0))
131:                         break;
132: 
133:                 /* Blink the LED (PIO0_2). */
134:                 gpio_toggle(PIO0_2);
135: 
136:                 if (++count >= 6) {
137:                         count = 0;
138: 
139:                         /* Pattern match (falling edge) */
140:                         pinint_set_bit_slice(0, 0, PININT_CONST_HIGH, false);
141:                         pinint_set_bit_slice(1, 0, PININT_STICKY_FALLING,
142:                                              true);
143: 
144:                         /* Enable patten match. */
145:                         pinint_enable_pattern_match();
146:                 }
147:                 break;
148:         default:
149:                 break;
150:         }
151: 
152:         /* Clear interrupt. */
153:         pinint_clear_interrupt(PININT0);
154: }
155: 
156: /* Pattern match */
157: void pinint1_isr(void)
158: {
159:         static int count;
160: 
161:         if (pinint_pattern_match_enabled()) {
162:                 if (count != 3)
163:                         /* Blink the LED (PIO0_2). */
164:                         gpio_toggle(PIO0_2);
165: 
166:                 count++;
167:                 if (count < 3) {
168:                         /* Falling edge */
169:                         pinint_set_bit_slice(1, 0, PININT_STICKY_FALLING,
170:                                              true);
171:                 } else if (count < 7) {
172:                         /* Rising edge */
173:                         pinint_set_bit_slice(1, 0, PININT_STICKY_RISING, true);
174:                 } else if (count < 13) {
175:                         /* Both edge */
176:                         pinint_set_bit_slice(1, 0, PININT_STICKY_BOTH, true);
177:                 } else {
178:                         count = 0;
179: 
180:                         /* Disable patten match. */
181:                         pinint_disable_pattern_match();
182: 
183:                         /* Clear interrupt */
184:                         pinint_clear_interrupt(PININT0);
185: 
186:                         /* Enable falling edge interrupt. */
187:                         pinint_set_interrupt_mode(PININT_FALLING, PININT0);
188:                 }
189:         }
190: }
191: 
192: int main(void)
193: {
194:         clock_setup();
195:         gpio_setup();
196:         pinint_setup();
197: 
198:         while (1)
199:                 ;
200: 
201:         return 0;
202: }
