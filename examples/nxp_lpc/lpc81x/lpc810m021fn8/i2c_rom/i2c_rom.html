041: /* Set LPC810 to 24 MHz. */
042: static void clock_setup(void)
043: {
044:         /* Set up PLL (Fclkin = 12MHz (IRC), Fclkout (Main clock) = 24MHz) */
045:         syscon_enable_pll(SYSCON_IRC, 2, 4);
046: 
047:         /* Select PLL as system clock source. */
048:         syscon_set_system_clock(SYSCON_PLL_OUT, 1);
049: 
050:         /* Enable USART clock (U_PCLK). */
051:         syscon_set_usart_clock(24000000, U_PCLK);
052: 
053: #ifdef FASTMODE
054:         /* Set IOCON glitch filter clock divider. */
055:         syscon_set_ioconclkdiv(0, 1);
056: #endif
057: }

059: static void gpio_setup(void)
060: {
061:         /* Enable GPIO clock. */
062:         // syscon_enable_clock(SYSCON_GPIO);
063: 
064:         /* Enable switch matrix clock. */
065:         // syscon_enable_clock(SYSCON_SWM);
066: 
067:         /* Enable IOCON clock. */
068:         syscon_enable_clock(SYSCON_IOCON);
069: 
070:         /* Prevent the I2C pins from internally floating. */
071:         gpio_config(GPIO_OUTPUT, GPIO_IO, PIO0_10 | PIO0_11);
072:         gpio_clear(PIO0_10 | PIO0_11);
073: 
074: #ifndef FASTMODE
075:         /* Set PIO0_2 to 'I2C SCL'. */
076:         gpio_config(GPIO_I2C_SCL, GPIO_OPENDRAIN | GPIO_HYST, PIO0_2);
077: 
078:         /* Set PIO0_3 to 'I2C SDA'. */
079:         gpio_config(GPIO_I2C_SDA, GPIO_OPENDRAIN | GPIO_HYST, PIO0_3);
080: #else
081:         /* Set PIO0_2 to 'I2C SCL'. */
082:         gpio_config(GPIO_I2C_SCL, GPIO_OPENDRAIN | GPIO_HYST |
083:                     GPIO_IOCONCLKDIV0 | GPIO_FILTER_2CLOCK, PIO0_2);
084: 
085:         /* Set PIO0_3 to 'I2C SDA'. */
086:         gpio_config(GPIO_I2C_SDA, GPIO_OPENDRAIN | GPIO_HYST |
087:                     GPIO_IOCONCLKDIV0 | GPIO_FILTER_2CLOCK, PIO0_3);
088: #endif
089: 
090:         /* Set PIO0_4 to 'USART0 TXD'. */
091:         gpio_config(GPIO_U0_TXD, GPIO_HYST, PIO0_4);
092: }

104: static void i2c_setup(void)
105: {
106:         /* Enable I2C clock. */
107:         syscon_enable_clock(SYSCON_I2C);
108: 
109: #ifndef FASTMODE
110:         /* Set fSCL to 100kHz (tHIGH = 5 usec, tLOW = 5 usec). */
111:         /* Standard-mode */
112:         i2c_set_scl(24, 5, 5);
113: #else
114:         /* Set fSCL to 400kHz (tHIGH = 1 usec, tLOW = 1.5 usec). */
115:         /* Fast-mode */
116:         i2c_set_scl(12, 2, 3);
117: #endif
118: 
119:         /* Enable I2C master function */
120:         i2c_enable_master(false);
121: }

123: static void print_hex(int c)
124: {
125:         int d;
126: 
127:         d = (c >> 4) & 0xf;
128:         if (d > 9)
129:                 d += 'a' - 10;
130:         else
131:                 d += '0';
132:         usart_send_blocking(USART0, d);
133:         d = c & 0xf;
134:         if (d > 9)
135:                 d += 'a' - 10;
136:         else
137:                 d += '0';
138:         usart_send_blocking(USART0, d);
139: }

155: static void print_int(int d)
156: {
157:         bool zerosup;
158:         int i;
159: 
160:         if (d < 0) {
161:                 usart_send_blocking(USART0, '-');
162:                 d = -d;
163:         }
164:         zerosup = true;
165:         for (i = 1000000000; i > 0; i /= 10) {
166:                 if (d / i || i == 1 || !zerosup) {
167:                         usart_send_blocking(USART0, d / i + '0');
168:                         d %= i;
169:                         zerosup = false;
170:                 }
171:         }
172: }

174: static void xprintf(char *p, int arg0, int arg1, int arg2)
175: {
176:         int arg[3];
177:         int i;
178: 
179:         arg[0] = arg0;
180:         arg[1] = arg1;
181:         arg[2] = arg2;
182:         i = 0;
183:         while (*p != '\0') {
184:                 if (*p == '%') {
185:                         switch (*++p) {
186:                         case '%':
187:                                 usart_send_blocking(USART0, *p++);
188:                                 break;
189:                         case 'x':
190:                                 if (i < 3)
191:                                         print_hex(arg[i++]);
192:                                 p++;
193:                                 break;
194:                         case 'h':
195:                                 if (i < 3)
196:                                         print_hex_half_word(arg[i++]);
197:                                 p++;
198:                                 break;
199:                         case 'w':
200:                                 if (i < 3)
201:                                         print_hex_word(arg[i++]);
202:                                 p++;
203:                                 break;
204:                         case 'd':
205:                                 if (i < 3)
206:                                         print_int(arg[i++]);
207:                                 p++;
208:                                 break;
209:                         default:
210:                                 usart_send_blocking(USART0, '?');
211:                                 p++;
212:                                 break;
213:                         }
214:                 } else {
215:                         usart_send_blocking(USART0, *p++);
216:                 }
217:         }
218: }

228: static int wait_i2c_state(int state)
229: {
230:         exp_state = state;
231:         do {
232:                 status = i2c_get_status(I2C_MSTPENDING | I2C_MSTSTATE);
233:         } while (!(status & I2C_MSTPENDING));
234: 
235:         if (state != (status & I2C_MSTSTATE))
236:                 return status & I2C_MSTSTATE;
237: 
238:         return 0;
239: }

241: /*
242:  * Byte Write 
243:  *
244:  * |S|10100000|A|xxx.....|A|........|A|........|A|P|
245:  *    Slave      Address    Address    Data
246:  *    Address    High Byte  Low Byte
247:  */
248: static int rom_write_byte(int addr, char data)
249: {
250:         int r;
251: 
252:         /* Idle */
253:         if ((r = wait_i2c_state(I2C_MST_IDLE)))
254:                 return r;
255: 
256:         do {
257:                 /* Send START + I2C Slave address. */
258:                 i2c_master_send_addr(ROM_SLA);
259: 
260:                 /* Receive ACK. */
261:                 if ((r = wait_i2c_state(I2C_MST_TX_RDY))) {
262:                         if (r != I2C_MST_NACK_ADDR)
263:                                 return r;
264:                 }
265:         } while (r == I2C_MST_NACK_ADDR);
266: 
267:         /* Send ROM address high byte. */
268:         i2c_master_send(addr >> 8);
269: 
270:         /* Receive ACK. */
271:         if ((r = wait_i2c_state(I2C_MST_TX_RDY)))
272:                 return r;
273: 
274:         /* Send ROM address low byte. */
275:         i2c_master_send(addr & 0xff);
276: 
277:         /* Receive ACK. */
278:         if ((r = wait_i2c_state(I2C_MST_TX_RDY)))
279:                 return r;
280: 
281:         /* Send data. */
282:         i2c_master_send(data);
283: 
284:         /* Receive ACK. */
285:         if ((r = wait_i2c_state(I2C_MST_TX_RDY)))
286:                 return r;
287: 
288:         /* Send STOP. */
289:         i2c_stop();
290: 
291:         /* Idle */
292:         if ((r = wait_i2c_state(I2C_MST_IDLE)))
293:                 return r;
294: 
295:         return 0;
296: }

298: /*
299:  * Page Write 
300:  *
301:  * |S|10100000|A|xxx.....|A|........|A|........|A|~|........|A|P|
302:  *    Slave      Address    Address    Data         Data
303:  *    Address    High Byte  Low Byte   Byte 0       Byte 31
304:  */
305: static int rom_write_page(int addr, char *data)
306: {
307:         int r;
308:         int i;
309: 
310:         /* Idle */
311:         if ((r = wait_i2c_state(I2C_MST_IDLE)))
312:                 return r;
313: 
314:         do {
315:                 /* Send START + I2C Slave address. */
316:                 i2c_master_send_addr(ROM_SLA);
317: 
318:                 /* Receive ACK. */
319:                 if ((r = wait_i2c_state(I2C_MST_TX_RDY))) {
320:                         if (r != I2C_MST_NACK_ADDR)
321:                                 return r;
322:                 }
323:         } while (r == I2C_MST_NACK_ADDR);
324: 
325:         /* Send ROM address high byte. */
326:         i2c_master_send(addr >> 8);
327: 
328:         /* Receive ACK. */
329:         if ((r = wait_i2c_state(I2C_MST_TX_RDY)))
330:                 return r;
331: 
332:         /* Send ROM address low byte. */
333:         i2c_master_send(addr & 0xff);
334: 
335:         for (i = 0; i < PAGE_SIZE; i++) {
336:                 /* Receive ACK. */
337:                 if ((r = wait_i2c_state(I2C_MST_TX_RDY)))
338:                         return r;
339: 
340:                 /* Send data. */
341:                 i2c_master_send(*data++);
342:         }
343: 
344:         /* Receive ACK. */
345:         if ((r = wait_i2c_state(I2C_MST_TX_RDY)))
346:                 return r;
347: 
348:         /* Send STOP. */
349:         i2c_stop();
350: 
351:         /* Idle */
352:         if ((r = wait_i2c_state(I2C_MST_IDLE)))
353:                 return r;
354: 
355:         return 0;
356: }

358: /*
359:  * Random Read / Sequential Read
360:  *
361:  * |S|10100000|A|xxx.....|A|........|A|S|10100001|A|........|A|~|........|N|P|
362:  *    Slave      Address    Address      Slave      Data n       Data n + x
363:  *    Address    High Byte  Low Byte     Address
364:  */
365: static int rom_read(int addr, char *data, int nbyte)
366: {
367:         int r;
368:         int i;
369: 
370:         /* Idle */
371:         if ((r = wait_i2c_state(I2C_MST_IDLE)))
372:                 return r;
373: 
374:         do {
375:                 /* Send START + I2C Slave address (Write). */
376:                 i2c_master_send_addr(ROM_SLA);
377: 
378:                 /* Receive ACK. */
379:                 if ((r = wait_i2c_state(I2C_MST_TX_RDY))) {
380:                         if (r != I2C_MST_NACK_ADDR)
381:                                 return r;
382:                 }
383:         } while (r == I2C_MST_NACK_ADDR);
384: 
385:         /* Send ROM address high byte. */
386:         i2c_master_send(addr >> 8);
387: 
388:         /* Receive ACK. */
389:         if ((r = wait_i2c_state(I2C_MST_TX_RDY)))
390:                 return r;
391: 
392:         /* Send ROM address low byte. */
393:         i2c_master_send(addr & 0xff);
394: 
395:         /* Receive ACK. */
396:         if ((r = wait_i2c_state(I2C_MST_TX_RDY)))
397:                 return r;
398: 
399:         /* Send START + I2C Slave address (Read). */
400:         i2c_master_send_addr(ROM_SLA | 1);
401: 
402:         for (i = 0; i < nbyte - 1; i++) {
403:                 /* Receive data. */
404:                 if ((r = wait_i2c_state(I2C_MST_RX_RDY)))
405:                         return r;
406: 
407:                 /* Get data and send ACK. */
408:                 *data++ = i2c_master_recv();
409:         }
410: 
411:         /* Receive data. */
412:         if ((r = wait_i2c_state(I2C_MST_RX_RDY)))
413:                 return r;
414: 
415:         /* Get data. */
416:         *data = i2c_master_recv_last();
417: 
418:         /* Send NACK + STOP. */
419:         i2c_stop();
420: 
421:         /* Idle */
422:         if ((r = wait_i2c_state(I2C_MST_IDLE)))
423:                 return r;
424: 
425:         return 0;
426: }

428: int main(void)
429: {
430:         char start_data;
431:         int addr;
432:         char data;
433:         int i;
434:         int j;
435:         char buf[256];
436:         int n;
437:         int m;
438: 
439:         clock_setup();
440:         gpio_setup();
441:         usart_setup();
442:         i2c_setup();
443: 
444:         xputs("i2c_rom");
445: 
446:         xputs("Random read");
447:         if (rom_read(0, &start_data, 1)) {
448:                 xputs("ROM read error");
449:                 goto i2c_error;
450:         }
451:         start_data++;
452:         xprintf("data = 0x%x\r\n", start_data, 0, 0);
453: 
454:         /* Write data. */
455:         addr = 0;
456:         data = start_data;
457:         /* Byte */
458:         xputs("Byte Write");
459:         for (i = 0; i < PAGE_SIZE; i++) {
460:                 if (rom_write_byte(addr++, data++)) {
461:                         xputs("ROM write error");
462:                         goto i2c_error;
463:                 }
464:         }
465:         /* Page */
466:         xputs("Page Write");
467:         for (i = 0; i < ROM_SIZE / PAGE_SIZE - 1; i++) {
468:                 for (j = 0; j < PAGE_SIZE; j++)
469:                         buf[j] = data++;
470:                 if (rom_write_page(addr, buf)) {
471:                         xputs("ROM write error");
472:                         goto i2c_error;
473:                 }
474:                 addr += PAGE_SIZE;
475:         }
476: 
477:         /* Read and verify data. */
478:         xputs("Sequential Read");
479:         while (1) {
480:                 for (n = 1; n < (int)sizeof(buf); n++) {
481:                         xprintf("size = %d\r\n", n, 0, 0);
482: 
483:                         addr = 0;
484:                         data = start_data;
485:                         for (i = 0; i < ROM_SIZE / n; i++) {
486:                                 if (rom_read(addr, buf, n)) {
487:                                         xputs("ROM read error");
488:                                         goto i2c_error;
489:                                 }
490:                                 for (j = 0; j < n; j++) {
491:                                         if (buf[j] != data)
492:                                                 goto data_error;
493:                                         data++;
494:                                 }
495:                                 addr += n;
496:                         }
497: 
498:                         m = ROM_SIZE % n;
499:                         if (m) {
500:                                 if (rom_read(addr, buf, m)) {
501:                                         xputs("ROM read error");
502:                                         goto i2c_error;
503:                                 }
504:                                 for (j = 0; j < m; j++) {
505:                                         if (buf[j] != data)
506:                                                 goto data_error;
507:                                         data++;
508:                                 }
509:                         }
510:                 }
511:         }
512: 
513: i2c_error:
514:         xprintf("I2C error: status = 0x%w  expected state = 0x%x\r\n",
515:                 status, exp_state, 0);
516:         while (1)
517:                 ;
518: 
519: data_error:
520:         xprintf("data error: addr = 0x%h  data = 0x%x (0x%x)\r\n",
521:                 addr + j, buf[j], data);
522:         while (1)
523:                 ;
524: 
525:         return 0;
526: }
