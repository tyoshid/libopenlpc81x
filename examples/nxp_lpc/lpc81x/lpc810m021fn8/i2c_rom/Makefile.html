18: NAME            = i2c_rom

20: OUTFILES        = $(NAME).bin $(NAME).list
21: LDSCRIPT        = ldscripts/lpc810.x
22: LDSPECS         =
23: LIBDIR          ?= ../../../../..
24: 
25: CC              = arm-none-eabi-gcc
26: LD              = arm-none-eabi-gcc
27: OBJCOPY         = arm-none-eabi-objcopy
28: OBJDUMP         = arm-none-eabi-objdump
29: SIZE            = arm-none-eabi-size

31: ARCHFLAGS       = -mthumb -mcpu=cortex-m0plus
32: CFLAGS          = -MMD -Os \
33:                   -Wall -Wextra -Wimplicit-function-declaration \
34:                   -Wredundant-decls -Wstrict-prototypes -Wundef \
35:                   -I$(LIBDIR)/include -I$(LIBDIR)/include/nxp_lpc/lpc81x \
36:                   -fno-common -fstack-usage $(ARCHFLAGS)
37: LDFLAGS         = -L$(LIBDIR)/lib/nxp_lpc/lpc81x -llpc81x \
38:                   -T $(LDSCRIPT) -nostartfiles $(LDSPECS) \
39:                   -Wl,--gc-sections -Wl,-Map=$(NAME).map -Wl,--cref \
40:                   $(ARCHFLAGS)
41: 
42: USARTUTIL       ?= $(LIBDIR)/tools/nxp_lpc/lpc81x/usart-util/usart-util
43: ifneq ($(strip $(USARTUTIL_DEVICE)),)
44: USARTUTIL       += -d $(USARTUTIL_DEVICE)
45: endif

47: .PHONY: all clean flash
48:

51: %.bin: %.elf
52:         echo "  $@"
53:         $(OBJCOPY) -O binary $< $@
54: 
55: %.hex: %.elf
56:         echo "  $@"
57:         $(OBJCOPY) -O ihex $< $@
58: 
59: %.srec: %.elf
60:         echo "  $@"
61:         $(OBJCOPY) -O srec $< $@
62: 
63: %.list: %.elf
64:         echo "  $@"
65:         $(OBJDUMP) -d $< > $@
66:         $(OBJDUMP) -t $< >> $@
67: 
68: $(NAME).elf: $(OBJS)
69:         echo "  $@"
70:         $(LD) -o $@ $(OBJS) $(LDFLAGS)
71:         $(SIZE) $(NAME).elf
72: 
73: %.o: %.c
74:         echo "  $<"
75:         $(CC) $(CFLAGS) -o $@ -c $<
