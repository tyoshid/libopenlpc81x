020: #include <syscon.h>
021: #include <gpio.h>
022: #include <usart.h>

025: #define FLASHSIZE       4096
026: 
027: /* USART clock frequency */
028: #define U_PCLK  (16 * 115200)
029: 
030: /* Set LPC810 to 24 MHz. */
031: static void clock_setup(void)
032: {
033:         /* Set up PLL (Fclkin = 12MHz (IRC), Fclkout (Main clock) = 24MHz) */
034:         syscon_enable_pll(SYSCON_IRC, 2, 4);
035: 
036:         /* Select PLL as system clock source. */
037:         syscon_set_system_clock(SYSCON_PLL_OUT, 1);
038: 
039:         /* Enable USART clock (U_PCLK). */
040:         syscon_set_usart_clock(24000000, U_PCLK);
041: }
042: 
043: static void gpio_setup(void)
044: {
045:         /* Enable GPIO clock. */
046:         // syscon_enable_clock(SYSCON_GPIO);
047: 
048:         /* Enable switch matrix clock. */
049:         // syscon_enable_clock(SYSCON_SWM);
050: 
051:         /* Enable IOCON clock. */
052:         syscon_enable_clock(SYSCON_IOCON);
053: 
054:         /* Prevent the pins from internally floating. */
055:         gpio_config(GPIO_OUTPUT, GPIO_IO, PIO0_10 | PIO0_11);
056:         gpio_clear(PIO0_10 | PIO0_11);
057: 
058:         /* Set PIO0_2 and PIO0_3 to 'output push-pull' (LED). */
059:         gpio_config(GPIO_OUTPUT, 0, PIO0_2 | PIO0_3);
060: 
061:         /* Set PIO0_4 to 'USART0 TXD'. */
062:         gpio_config(GPIO_U0_TXD, GPIO_HYST, PIO0_4);
063: }
064: 
065: static void usart_setup(void)
066: {
067:         /* Enable USART0 clock. */
068:         syscon_enable_clock(SYSCON_UART0);
069: 
070:         /* Set up USART0. */
071:         usart_init(USART0, U_PCLK, 115200, 8, 1, USART_PARITY_NONE,
072:                    USART_FLOW_NONE);
073: }

075: static void print_hex(int c)
076: {
077:         int d;
078: 
079:         d = (c >> 4) & 0xf;
080:         if (d > 9)
081:                 d += 'a' - 10;
082:         else
083:                 d += '0';
084:         usart_send_blocking(USART0, d);
085:         d = c & 0xf;
086:         if (d > 9)
087:                 d += 'a' - 10;
088:         else
089:                 d += '0';
090:         usart_send_blocking(USART0, d);
091: }
092: 
093: static void print_hex_half_word(int w)
094: {
095:         print_hex((w >> 8) & 0xff);
096:         print_hex(w & 0xff);
097: }
098: 
099: static void print_hex_word(int w)
100: {
101:         print_hex((w >> 24) & 0xff);
102:         print_hex((w >> 16) & 0xff);
103:         print_hex((w >> 8) & 0xff);
104:         print_hex(w & 0xff);
105: }
106: 
107: static void print_int(int d)
108: {
109:         bool zerosup;
110:         int i;
111: 
112:         if (d < 0) {
113:                 usart_send_blocking(USART0, '-');
114:                 d = -d;
115:         }
116:         zerosup = true;
117:         for (i = 1000000000; i > 0; i /= 10) {
118:                 if (d / i || i == 1 || !zerosup) {
119:                         usart_send_blocking(USART0, d / i + '0');
120:                         d %= i;
121:                         zerosup = false;
122:                 }
123:         }
124: }

126: static void xprintf(char *p, int arg0, int arg1, int arg2)
127: {
128:         int arg[3];
129:         int i;
130: 
131:         arg[0] = arg0;
132:         arg[1] = arg1;
133:         arg[2] = arg2;
134:         i = 0;
135:         while (*p != '\0') {
136:                 if (*p == '%') {
137:                         switch (*++p) {
138:                         case '%':
139:                                 usart_send_blocking(USART0, *p++);
140:                                 break;
141:                         case 'x':
142:                                 if (i < 3)
143:                                         print_hex(arg[i++]);
144:                                 p++;
145:                                 break;
146:                         case 'h':
147:                                 if (i < 3)
148:                                         print_hex_half_word(arg[i++]);
149:                                 p++;
150:                                 break;
151:                         case 'w':
152:                                 if (i < 3)
153:                                         print_hex_word(arg[i++]);
154:                                 p++;
155:                                 break;
156:                         case 'd':
157:                                 if (i < 3)
158:                                         print_int(arg[i++]);
159:                                 p++;
160:                                 break;
161:                         default:
162:                                 usart_send_blocking(USART0, '?');
163:                                 p++;
164:                                 break;
165:                         }
166:                 } else {
167:                         usart_send_blocking(USART0, *p++);
168:                 }
169:         }
170: }

172: static void xputs(char *p)
173: {
174:         while (*p != '\0')
175:                 usart_send_blocking(USART0, *p++);
176:         usart_send_blocking(USART0, '\r');
177:         usart_send_blocking(USART0, '\n');
178: }

180: static int signature(int start, int stop)
181: {
182:         unsigned int *p;
183:         unsigned int s;
184:         int i;
185:         unsigned int n;
186: 
187:         p = (unsigned int *)start;
188:         s = 0;
189:         for (i = start; i < stop; i += 4) {
190:                 n = *p++ ^ (s >> 1);
191:                 if (s & (1 << 0))
192:                         n ^= 0x80000000;
193:                 if (s & (1 << 10))
194:                         n ^= 0x80000000;
195:                 if (s & (1 << 30))
196:                         n ^= 0x80000000;
197:                 if (s & (1 << 31))
198:                         n ^= 0x80000000;
199:                 s = n;
200:         }
201:         return s;
202: }

204: int main(void)
205: {
206:         int i;
207:         int j;
208:         int r;
209:         int s;
210: 
211:         clock_setup();
212:         gpio_setup();
213:         usart_setup();
214: 
215:         xputs("Flash controller");
216: 
217:         while (1) {
218:                 for (i = 0; i < FLASHSIZE; i += 4) {
219:                         /* LED1 on/off */
220:                         gpio_toggle(PIO0_2);
221: 
222:                         for (j = i + 3; j < FLASHSIZE; j += 4) {
223:                                 r = flashcon_generate_signature(i, j);
224:                                 s = signature(i, j);
225:                                 if (r != s) {
226:                                         xprintf("Fail %w %w ", i, j, 0);
227:                                         xprintf("%w (%w)\r\n", r, s, 0);
228:                                         /* LED2 on */
229:                                         gpio_set(PIO0_3);
230:                                 }
231:                         }
232:                 }
233:         }
234: 
235:         return 0;
236: }
