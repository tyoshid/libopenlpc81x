020: #include <syscon.h>
021: #include <gpio.h>
022: #include <nvic.h>
023: #include <sct.h>
024: 
025: #define PCLK    24000000
026: 
027: const int freq[] = {
028:         1047,
029:         1175,
030:         1319,
031:         1397,
032:         1568,
033:         1760,
034:         1976,
035:         2093
036: };
037: 
038: volatile int findex;
039: volatile int count;
040: 
041: /* Set LPC810 to 24 MHz. */
042: static void clock_setup(void)
043: {
044:         /* Set up PLL (Fclkin = 12MHz (IRC), Fclkout (Main clock) = 24MHz) */
045:         syscon_enable_pll(SYSCON_IRC, 2, 4);
046: 
047:         /* Select PLL as system clock source. */
048:         syscon_set_system_clock(SYSCON_PLL_OUT, 1);
049: }
050: 
051: static void gpio_setup(void)
052: {
053:         /* Enable GPIO clock. */
054:         // syscon_enable_clock(SYSCON_GPIO);
055: 
056:         /* Enable switch matrix clock. */
057:         // syscon_enable_clock(SYSCON_SWM);
058: 
059:         /* Enable IOCON clock. */
060:         syscon_enable_clock(SYSCON_IOCON);
061: 
062:         /* Prevent the I2C pins from internally floating. */
063:         gpio_config(GPIO_OUTPUT, GPIO_IO, PIO0_10 | PIO0_11);
064:         gpio_clear(PIO0_10 | PIO0_11);
065: 
066:         /* Set PIO0_2 to 'SCT output 0'. */
067:         gpio_config(GPIO_CTOUT0, 0, PIO0_2);
068: 
069:         /* Set PIO0_3 to 'SCT output 1'. */
070:         gpio_config(GPIO_CTOUT1, 0, PIO0_3);
071: }
072: 
073: static void sct_setup(void)
074: {
075:         /* Enable SCT clock. */
076:         syscon_enable_clock(SYSCON_SCT);
077: 
078:         /* Enable the SCT interrupt. */
079:         nvic_enable_irq(NVIC_SCT);
080: 
081:         /* Set counter to 32-bit. */
082:         sct_config(SCT_CLOCK_BUS, 0, SCT_UNIFY | SCT_AUTOLIMIT);
083: 
084:         /* Match0: limit */
085:         sct_set_match_and_reload(0, PCLK / freq[findex] - 1);
086: 
087:         /* Duration (1 sec)*/
088:         count = freq[findex];
089: 
090:         /* Match1: low threshold */
091:         sct_set_match_and_reload(1, 0);
092: 
093:         /* Match2: high threshold (duty 50%) */
094:         sct_set_match_and_reload(2, PCLK / freq[findex] / 2);
095: 
096:         /* Event0: match1 -> output0 high */
097:         sct_setup_event(0, 1, 0, SCT_STATE0 | SCT_MATCH_ONLY,
098:                         SCT_EV_OUT0_SET, 0);
099: 
100:         /* Event1: match2 -> output0 low */
101:         sct_setup_event(1, 2, 0, SCT_STATE0 | SCT_MATCH_ONLY,
102:                         SCT_EV_OUT0_CLR, 0);
103: 
104:         /* Event2: match1 -> output1 low */
105:         sct_setup_event(2, 1, 0, SCT_STATE0 | SCT_MATCH_ONLY,
106:                         SCT_EV_OUT1_CLR, 0);
107: 
108:         /* Event3: match2 -> output1 high */
109:         sct_setup_event(3, 2, 0, SCT_STATE0 | SCT_MATCH_ONLY,
110:                         SCT_EV_OUT1_SET, 0);
111: 
112:         /* Event4: match1 -> interrupt */
113:         sct_setup_event(4, 1, 0, SCT_STATE0 | SCT_MATCH_ONLY, SCT_EV_INT, 0);
114: 
115:         /* Start timer. */
116:         sct_start_counter();
117: }
118: 
119: void sct_isr(void)
120: {
121:         if (sct_get_interrupt_mask(SCT_EVENT4) &&
122:             sct_get_interrupt_status(SCT_EVENT4)) {
123:                 if (--count == 0) {
124:                         if (++findex == sizeof(freq) / sizeof(int))
125:                                 findex = 0;
126: 
127:                         /* Match0: limit */
128:                         sct_set_match_reload(0, PCLK / freq[findex] - 1);
129: 
130:                         /* Duration (1 sec)*/
131:                         count = freq[findex];
132: 
133:                         /* Match2: high threshold (duty 50%) */
134:                         sct_set_match_reload(2, PCLK / freq[findex] / 2);
135:                 }
136: 
137:                 /* Clear interrupt. */
138:                 sct_clear_interrupt(SCT_EVENT4);
139:         }
140: }
141: 
142: int main(void)
143: {
144:         clock_setup();
145:         gpio_setup();
146:         sct_setup();
147: 
148:         while (1)
149:                 ;
150: 
151:         return 0;
152: }
