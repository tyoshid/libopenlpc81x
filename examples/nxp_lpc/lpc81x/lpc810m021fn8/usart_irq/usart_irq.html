020: #include <syscon.h>
021: #include <gpio.h>
022: #include <nvic.h>
023: #include <usart.h>
024: 
025: /* USART clock frequency */
026: #define U_PCLK  (16 * 115200)
027: 
028: /* Queue size */
029: #define QUEUESIZE       8
030: 
031: /* Set LPC810 to 30 MHz. */
032: static void clock_setup(void)
033: {
034:         /*
035:          * Select IRC (12MHz) as PLL source (Fclkin).
036:          * Main clock (Fclkout) is 60MHz (< 100MHz).
037:          * M = Fclkout / Fclkin = 5
038:          * FCCO = 2 * P * Fclkout (156MHz <= FCCO <= 320MHz)
039:          * P = 2 (FCCO = 240MHz)
040:          */
041:         syscon_enable_pll(SYSCON_IRC, 5, 2);
042: 
043:         /*
044:          * Select PLL as system clock source.
045:          * Main clock:   60MHz
046:          * System clock: 30MHz
047:          */
048:         syscon_set_system_clock(SYSCON_PLL_OUT, 2);
049: 
050:         /* Enable USART clock (U_PCLK). */
051:         syscon_set_usart_clock(60000000, U_PCLK);
052: }
053: 
054: static void gpio_setup(void)
055: {
056:         /* Enable GPIO clock. */
057:         // syscon_enable_clock(SYSCON_GPIO);
058: 
059:         /* Enable switch matrix clock. */
060:         // syscon_enable_clock(SYSCON_SWM);
061: 
062:         /* Enable IOCON clock. */
063:         syscon_enable_clock(SYSCON_IOCON);
064: 
065:         /* Prevent the pins from internally floating. */
066:         gpio_config(GPIO_OUTPUT, GPIO_IO, PIO0_10 | PIO0_11);
067:         gpio_clear(PIO0_10 | PIO0_11);
068: 
069:         /* Set PIO0_2 and PIO0_3 to 'output push-pull'. */
070:         gpio_config(GPIO_OUTPUT, 0, PIO0_2 | PIO0_3);
071: 
072:         /* Set PIO0_4 to 'USART0 TXD'. */
073:         gpio_config(GPIO_U0_TXD, GPIO_HYST, PIO0_4);
074: 
075:         /* Set PIO0_0 to 'USART0 RXD'. */
076:         gpio_config(GPIO_U0_RXD, GPIO_HYST, PIO0_0);
077: }
078: 
079: static void usart_setup(void)
080: {
081:         /* Enable USART0 clock. */
082:         syscon_enable_clock(SYSCON_UART0);
083: 
084:         /* Enable USART0 interrupt. */
085:         nvic_enable_irq(NVIC_UART0);
086: 
087:         /* Enable USART0 Receive interrupt. */
088:         usart_enable_interrupt(USART0, USART_RXRDY);
089: 
090:         /* Set up USART0. */
091:         usart_init(USART0, U_PCLK, 115200, 8, 1, USART_PARITY_NONE,
092:                    USART_FLOW_NONE);
093: }
094: 
095: void uart0_isr(void)
096: {
097:         int m;
098:         int s;
099: 
100:         /* Rx, Tx queue */
101:         static int queue[QUEUESIZE];
102:         static int head;        /* read pointer */
103:         static int tail;        /* write pointer */
104:         static int qlen;
105: 
106:         /* Get interrupt mask. */
107:         m = usart_get_interrupt_mask(USART0, USART_RXRDY | USART_TXRDY);
108: 
109:         /* Get interrupt status. */
110:         s = usart_get_interrupt_status(USART0, USART_RXRDY | USART_TXRDY);
111: 
112:         /* RxRDY */
113:         if (m & s & USART_RXRDY) {
114:                 if (qlen < QUEUESIZE) {
115:                         /* LED1 on/off */
116:                         gpio_toggle(PIO0_2);
117: 
118:                         /* Receive data. */
119:                         queue[tail++] = usart_recv(USART0);
120:                         tail %= QUEUESIZE;
121:                         qlen++;
122: 
123:                         /* Enable transmit interrupt. */
124:                         if (!(m & USART_TXRDY))
125:                                 usart_enable_interrupt(USART0, USART_TXRDY);
126:                 } else {
127:                         /* Overflow */
128:                         /* Clear interrupt. */
129:                         usart_recv(USART0);
130:                 }
131: 
132:         }
133: 
134:         /* TxRDY */
135:         if (m & s & USART_TXRDY) {
136:                 if (qlen) {
137:                         /* LED2 on/off */
138:                         gpio_toggle(PIO0_3);
139: 
140:                         /* Send data. */
141:                         usart_send(USART0, queue[head++]);
142:                         head %= QUEUESIZE;
143:                         qlen--;
144:                 } else {
145:                         /* No more data */
146:                         /* Disable transmit interrupt. */
147:                         usart_disable_interrupt(USART0, USART_TXRDY);
148:                 }
149:         }
150: }
151: 
152: int main(void)
153: {
154:         clock_setup();
155:         gpio_setup();
156:         usart_setup();
157: 
158:         while (1)
159:                 ;
160: 
161:         return 0;
162: }
