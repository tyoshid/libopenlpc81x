20: #include <syscon.h>
21: #include <gpio.h>
22: 
23: volatile int dont_delete_loop;
24: 
25: /* Set LPC810 to 24 MHz. */
26: static void clock_setup(void)
27: {
28:         /*
29:          * Select IRC (12MHz) as PLL source (Fclkin).
30:          * Main clock (Fclkout) is 24MHz (< 100MHz).
31:          * M = Fclkout / Fclkin = 2
32:          * FCCO = 2 * P * Fclkout (156MHz <= FCCO <= 320MHz)
33:          * P = 4 (FCCO = 192MHz)
34:          */
35:         syscon_enable_pll(SYSCON_IRC, 2, 4);
36: 
37:         /*
38:          * Select PLL as system clock source.
39:          * Main clock:   24MHz
40:          * System clock: 24MHz
41:          */
42:         syscon_set_system_clock(SYSCON_PLL_OUT, 1);
43: }
44: 
45: static void gpio_setup(void)
46: {
47:         /* Enable GPIO clock. */
48:         // syscon_enable_clock(SYSCON_GPIO);
49: 
50:         /* Enable switch matrix clock. */
51:         // syscon_enable_clock(SYSCON_SWM);
52: 
53:         /* Enable IOCON clock. */
54:         syscon_enable_clock(SYSCON_IOCON);
55: 
56:         /* Prevent the I2C pins from internally floating. */
57:         gpio_config(GPIO_OUTPUT, GPIO_IO, PIO0_10 | PIO0_11);
58:         gpio_clear(PIO0_10 | PIO0_11);
59: 
60:         /* Set PIO0_2 and PIO0_3 to 'output push-pull'. */
61:         gpio_config(GPIO_OUTPUT, 0, PIO0_2 | PIO0_3);
62: 
63:         /* LED1 on */
64:         gpio_set(PIO0_2);
65: 
66:         /* LED2 off */
67:         gpio_clear(PIO0_3);
68: }
69: 
70: int main(void)
71: {
72:         int i;
73: 
74:         clock_setup();
75:         gpio_setup();
76: 
77:         /* Blink the LEDs (PIO0_2 and PIO0_3). */
78:         while (1) {
79:                 /* Wait a bit. */
80:                 for (i = 0; i < 500000; i++)
81:                         dont_delete_loop;
82: 
83:                 /* LED on/off */
84:                 gpio_toggle(PIO0(2, 3));
85:         }
86: 
87:         return 0;
88: }
