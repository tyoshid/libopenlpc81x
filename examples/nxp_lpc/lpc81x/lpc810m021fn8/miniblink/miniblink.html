23: #include <gpio.h>
24: 
25: volatile int dont_delete_loop;
26: 
27: static void gpio_setup(void)
28: {
29:         /* Manually: */
30:         /* Enable IOCON clock. */
31:         // SYSCON_SYSAHBCLKCTRL |= SYSCON_SYSAHBCLKCTRL_IOCON;
32:         /* Using API functions: */
33:         syscon_enable_clock(SYSCON_IOCON);
34: 
35:         /* Prevent the I2C pins from internally floating. */
36:         // IOCON_PIO0_10 = IOCON_I2CMODE_IO;
37:         // IOCON_PIO0_11 = IOCON_I2CMODE_IO;
38:         // GPIO_DIR0 |= GPIO_DIR0_OUTPUT << 10 | GPIO_DIR0_OUTPUT << 11;
39:         // GPIO_CLR0 |= 1 << 10 | 1 << 11;
40:         gpio_config(GPIO_OUTPUT, GPIO_IO, PIO0_10 | PIO0_11);
41:         gpio_clear(PIO0_10 | PIO0_11);
42: 
43:         /* Manually: */
44:         /* Disable SWDIO. */
45:         // SWM_PINENABLE0 |= SWM_PINENABLE0_SWDIO_EN;
46:         /* Set IOCON to 'output pushpull'. */
47:         // IOCON_PIO0_2 = 0;
48:         /* Set PIO0_2 to output. */
49:         // GPIO_DIR0 = GPIO_DIR0_OUTPUT << 2;
50:         /* Using API functions: */
51:         gpio_config(GPIO_OUTPUT, 0, PIO0_2);
52: }
53: 
54: int main(void)
55: {
56:         int i;
57: 
58:         gpio_setup();
59: 
60:         /* Blink the LED (PIO0_2). */
61:         while (1) {
62:                 /* Manually: */
63:                 // GPIO_B2 = 1;                 /* LED on */
64:                 // for (i = 0; i < 500000; i++) /* Wait a bit. */
65:                 //      dont_delete_loop;
66:                 // GPIO_B2 = 0;                 /* LED off */
67:                 // for (i = 0; i < 500000; i++) /* Wait a bit. */
68:                 //      dont_delete_loop;
69: 
70:                 /* Using API functions gpio_set()/gpio_clear(): */
71:                 // gpio_set(PIO0_2);            /* LED on */
72:                 // for (i = 0; i < 500000; i++) /* Wait a bit. */
73:                 //      dont_delete_loop;
74:                 // gpio_clear(PIO0_2);          /* LED off */
75:                 // for (i = 0; i < 500000; i++) /* Wait a bit. */
76:                 //      dont_delete_loop;
77: 
78:                 /* Using API function gpio_toggle(): */
79:                 gpio_toggle(PIO0_2);            /* LED on/off */
80:                 for (i = 0; i < 500000; i++)    /* Wait a bit. */
81:                         dont_delete_loop;
82:         }
83: 
84:         return 0;
85: }
