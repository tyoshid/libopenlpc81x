020: #include <syscon.h>
021: #include <gpio.h>
022: #include <usart.h>
023: #include <flash_iap.h>
024: 
025: #define TEST_FLASH_ADDRESS      0x00000c00
026: #define TEST_SECTOR             3
027: #define SECTOR_SIZE             1024
028: #define TEST_DATA_SIZE          256
029: 
030: /* USART clock frequency */
031: #define U_PCLK  (16 * 115200)
032: 
033: volatile int dont_delete_loop;
034: 
035: /* Set LPC810 to 24 MHz. */
036: static void clock_setup(void)
037: {
038:         /* Set up PLL (Fclkin = 12MHz (IRC), Fclkout (Main clock) = 24MHz) */
039:         syscon_enable_pll(SYSCON_IRC, 2, 4);
040: 
041:         /* Select PLL as system clock source. */
042:         syscon_set_system_clock(SYSCON_PLL_OUT, 1);
043: 
044:         /* Enable USART clock (U_PCLK). */
045:         syscon_set_usart_clock(24000000, U_PCLK);
046: }
047: 
048: static void gpio_setup(void)
049: {
050:         /* Enable GPIO clock. */
051:         // syscon_enable_clock(SYSCON_GPIO);
052: 
053:         /* Enable switch matrix clock. */
054:         // syscon_enable_clock(SYSCON_SWM);
055: 
056:         /* Enable IOCON clock. */
057:         syscon_enable_clock(SYSCON_IOCON);
058: 
059:         /* Prevent the pins from internally floating. */
060:         gpio_config(GPIO_OUTPUT, GPIO_IO, PIO0_10 | PIO0_11);
061:         gpio_clear(PIO0_10 | PIO0_11);
062: 
063:         /* Set PIO0_2 to 'output push-pull' (LED). */
064:         gpio_config(GPIO_OUTPUT, 0, PIO0_2);
065: 
066:         /* Set PIO0_4 to 'USART0 TXD'. */
067:         gpio_config(GPIO_U0_TXD, GPIO_HYST, PIO0_4);
068: }
069: 
070: static void usart_setup(void)
071: {
072:         /* Enable USART0 clock. */
073:         syscon_enable_clock(SYSCON_UART0);
074: 
075:         /* Set up USART0. */
076:         usart_init(USART0, U_PCLK, 115200, 8, 1, USART_PARITY_NONE,
077:                    USART_FLOW_NONE);
078: }
079: 
080: static void print_hex(int c)
081: {
082:         int d;
083: 
084:         d = (c >> 4) & 0xf;
085:         if (d > 9)
086:                 d += 'a' - 10;
087:         else
088:                 d += '0';
089:         usart_send_blocking(USART0, d);
090:         d = c & 0xf;
091:         if (d > 9)
092:                 d += 'a' - 10;
093:         else
094:                 d += '0';
095:         usart_send_blocking(USART0, d);
096: }
097: 
098: static void print_hex_half_word(int w)
099: {
100:         print_hex((w >> 8) & 0xff);
101:         print_hex(w & 0xff);
102: }
103: 
104: static void print_hex_word(int w)
105: {
106:         print_hex((w >> 24) & 0xff);
107:         print_hex((w >> 16) & 0xff);
108:         print_hex((w >> 8) & 0xff);
109:         print_hex(w & 0xff);
110: }
111: 
112: static void print_int(int d)
113: {
114:         bool zerosup;
115:         int i;
116: 
117:         if (d < 0) {
118:                 usart_send_blocking(USART0, '-');
119:                 d = -d;
120:         }
121:         zerosup = true;
122:         for (i = 1000000000; i > 0; i /= 10) {
123:                 if (d / i || i == 1 || !zerosup) {
124:                         usart_send_blocking(USART0, d / i + '0');
125:                         d %= i;
126:                         zerosup = false;
127:                 }
128:         }
129: }

131: static void xprintf(char *p, int arg0, int arg1, int arg2)
132: {
133:         int arg[3];
134:         int i;
135: 
136:         arg[0] = arg0;
137:         arg[1] = arg1;
138:         arg[2] = arg2;
139:         i = 0;
140:         while (*p != '\0') {
141:                 if (*p == '%') {
142:                         switch (*++p) {
143:                         case '%':
144:                                 usart_send_blocking(USART0, *p++);
145:                                 break;
146:                         case 'x':
147:                                 if (i < 3)
148:                                         print_hex(arg[i++]);
149:                                 p++;
150:                                 break;
151:                         case 'h':
152:                                 if (i < 3)
153:                                         print_hex_half_word(arg[i++]);
154:                                 p++;
155:                                 break;
156:                         case 'w':
157:                                 if (i < 3)
158:                                         print_hex_word(arg[i++]);
159:                                 p++;
160:                                 break;
161:                         case 'd':
162:                                 if (i < 3)
163:                                         print_int(arg[i++]);
164:                                 p++;
165:                                 break;
166:                         default:
167:                                 usart_send_blocking(USART0, '?');
168:                                 p++;
169:                                 break;
170:                         }
171:                 } else {
172:                         usart_send_blocking(USART0, *p++);
173:                 }
174:         }
175: }
176: 
177: static void xputs(char *p)
178: {
179:         while (*p != '\0')
180:                 usart_send_blocking(USART0, *p++);
181:         usart_send_blocking(USART0, '\r');
182:         usart_send_blocking(USART0, '\n');
183: }

185: int main(void)
186: {
187:         IAP iap_entry = (IAP)IAP_LOCATION;
188:         unsigned int command_param[5];
189:         unsigned int status_result[4];
190:         char *p;
191:         char start_data;
192:         char data;
193:         int i;
194:         int j;
195:         char buf[TEST_DATA_SIZE] __attribute__ ((aligned(4)));
196: 
197:         clock_setup();
198:         gpio_setup();
199:         usart_setup();
200: 
201:         xputs("Flash IAP");
202: 
203:         /* Read Part Identification number */
204:         command_param[0] = IAP_READ_PART_ID;
205:         iap_entry(command_param, status_result);
206:         xprintf("Part ID = 0x%w\r\n", status_result[1], 0, 0);
207: 
208:         /* Read Boot code version number */
209:         command_param[0] = IAP_READ_BOOT_CODE_VERSION;
210:         iap_entry(command_param, status_result);
211:         xprintf("Boot code version = %d.%d\r\n",
212:                 (status_result[1] >> 8) & 0xff, status_result[1] & 0xff, 0);
213: 
214:         p = (char *)TEST_FLASH_ADDRESS;
215:         start_data = *p + 1;
216:         data = start_data;
217: 
218:         xputs("Blank check sector");
219:         command_param[0] = IAP_BLANK_CHECK_SECTOR;
220:         command_param[1] = TEST_SECTOR;
221:         command_param[2] = TEST_SECTOR;
222:         iap_entry(command_param, status_result);
223:         if (status_result[0] == IAP_SECTOR_NOT_BLANK) {
224:                 xputs("Prepare sector for write operation");
225:                 command_param[0] = IAP_PREPARE_SECTOR_FOR_WRITE_OPERATION;
226:                 command_param[1] = TEST_SECTOR;
227:                 command_param[2] = TEST_SECTOR;
228:                 iap_entry(command_param, status_result);
229:                 if (status_result[0] != IAP_CMD_SUCCESS) {
230:                         xprintf("IAP Prepare sector(s) for write operation "
231:                                 "command failed: %d\r\n", 
232:                                 status_result[0], 0, 0);
233:                         goto iap_error;
234:                 }
235: 
236:                 xputs("Erase sector");
237:                 command_param[0] = IAP_ERASE_SECTOR;
238:                 command_param[1] = TEST_SECTOR;
239:                 command_param[2] = TEST_SECTOR;
240:                 command_param[3] = 24000;
241:                 iap_entry(command_param, status_result);
242:                 if (status_result[0] != IAP_CMD_SUCCESS) {
243:                         xprintf("IAP Erase sector(s) command failed: %d\r\n", 
244:                                 status_result[0], 0, 0);
245:                         goto iap_error;
246:                 }
247: 
248:                 xputs("Blank check sector");
249:                 command_param[0] = IAP_BLANK_CHECK_SECTOR;
250:                 command_param[1] = TEST_SECTOR;
251:                 command_param[2] = TEST_SECTOR;
252:                 iap_entry(command_param, status_result);
253:                 if (status_result[0] != IAP_CMD_SUCCESS) {
254:                         xprintf("IAP Blank check sector(s) command failed: "
255:                                 "%d\r\n", 
256:                                 status_result[0], 0, 0);
257:                         goto iap_error;
258:                 }
259:         } else if (status_result[0] != IAP_CMD_SUCCESS) {
260:                 xprintf("IAP Blank check sector(s) command failed: %d\r\n", 
261:                         status_result[0], 0, 0);
262:                 goto iap_error;
263:         }
264: 
265:         for (i = 0; i < SECTOR_SIZE / TEST_DATA_SIZE; i++) {
266:                 /* Generate test data. */
267:                 for (j = 0; j < TEST_DATA_SIZE; j++)
268:                         buf[j] = data++;
269: 
270:                 xputs("Prepare sector for write operation");
271:                 command_param[0] = IAP_PREPARE_SECTOR_FOR_WRITE_OPERATION;
272:                 command_param[1] = TEST_SECTOR;
273:                 command_param[2] = TEST_SECTOR;
274:                 iap_entry(command_param, status_result);
275:                 if (status_result[0] != IAP_CMD_SUCCESS) {
276:                         xprintf("IAP Prepare sector(s) for write operation "
277:                                 "command failed: %d\r\n", 
278:                                 status_result[0], 0, 0);
279:                         goto iap_error;
280:                 }
281: 
282:                 xputs("Copy RAM to flash");
283:                 command_param[0] = IAP_COPY_RAM_TO_FLASH;
284:                 command_param[1] = TEST_FLASH_ADDRESS + TEST_DATA_SIZE * i;
285:                 command_param[2] = (unsigned int)buf;
286:                 command_param[3] = TEST_DATA_SIZE;
287:                 command_param[4] = 24000;
288:                 iap_entry(command_param, status_result);
289:                 if (status_result[0] != IAP_CMD_SUCCESS) {
290:                         xprintf("IAP Prepare sector(s) for write operation "
291:                                 "command failed: %d\r\n", 
292:                                 status_result[0], 0, 0);
293:                         goto iap_error;
294:                 }
295: 
296:                 xputs("Compare");
297:                 command_param[0] = IAP_COMPARE;
298:                 command_param[1] = TEST_FLASH_ADDRESS + TEST_DATA_SIZE * i;
299:                 command_param[2] = (unsigned int)buf;
300:                 command_param[3] = TEST_DATA_SIZE;
301:                 iap_entry(command_param, status_result);
302:                 if (status_result[0] != IAP_CMD_SUCCESS) {
303:                         xprintf("IAP Compare command failed: %d\r\n", 
304:                                 status_result[0], 0, 0);
305:                         goto iap_error;
306:                 }
307:         }
308: 
309:         p = (char *)TEST_FLASH_ADDRESS;
310:         data = start_data;
311:         for (i = 0; i < SECTOR_SIZE; i++) {
312:                 if (*p != data) {
313:                         xprintf("Verify error: 0x%w 0x%x (0x%x)\r\n",
314:                                 (int)p, *p, data);
315:                         goto iap_error;
316:                 }
317:                 p++;
318:                 data++;
319:         }
320: 
321:         xputs("Reinvoke ISP");
322:         /* LED on */
323:         gpio_set(PIO0_2);
324:         command_param[0] = IAP_REINVOKE_ISP;
325:         iap_entry(command_param, status_result);
326: 
327: iap_error:
328:         while (1) {
329:                 /* LED on/off */
330:                 gpio_toggle(PIO0_2);
331: 
332:                 /* Wait a bit. */
333:                 for (i = 0; i < 1000000; i++)
334:                         dont_delete_loop;
335:         }
336: 
337:         return 0;
338: }
