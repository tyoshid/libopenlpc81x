020: #include <syscon.h>
021: #include <gpio.h>
022: #include <sct.h>
023: #include <mrt.h>
024: #include <acmp.h>
025: 
026: /* Set LPC810 to 24 MHz. */
027: static void clock_setup(void)
028: {
029:         /* Set up PLL (Fclkin = 12MHz (IRC), Fclkout (Main clock) = 24MHz) */
030:         syscon_enable_pll(SYSCON_IRC, 2, 4);
031: 
032:         /* Select PLL as system clock source. */
033:         syscon_set_system_clock(SYSCON_PLL_OUT, 1);
034: }
035: 
036: static void gpio_setup(void)
037: {
038:         /* Enable GPIO clock. */
039:         // syscon_enable_clock(SYSCON_GPIO);
040: 
041:         /* Enable switch matrix clock. */
042:         // syscon_enable_clock(SYSCON_SWM);
043: 
044:         /* Enable IOCON clock. */
045:         syscon_enable_clock(SYSCON_IOCON);
046: 
047:         /* Prevent the pins from internally floating. */
048:         gpio_config(GPIO_OUTPUT, GPIO_IO, PIO0_10 | PIO0_11);
049:         gpio_clear(PIO0_10 | PIO0_11);
050: 
051:         /* Set PIO0_2 to 'SCT output 0'. */
052:         gpio_config(GPIO_CTOUT0, 0, PIO0_2);
053: 
054:         /* Set PIO0_0 to 'ACMP input 1'. */
055:         gpio_config(GPIO_ACMP_I1, 0, PIO0_0);
056: }
057: 
058: static void sct_setup(void)
059: {
060:         /* Enable SCT clock. */
061:         syscon_enable_clock(SYSCON_SCT);
062: 
063:         /* Set counter to 32-bit. */
064:         sct_config(SCT_CLOCK_BUS, 0, SCT_UNIFY | SCT_AUTOLIMIT);
065: 
066:         /* Match0: limit */
067:         /* 100Hz */
068:         sct_set_match_and_reload(0, 24000000 / 100 - 1);
069: 
070:         /* Match1: PWM low threshold */
071:         sct_set_match_and_reload(1, 0);
072: 
073:         /* Match2: PWM high threshold */
074:         sct_set_match_and_reload(2, 24000000 / 100 / 32);
075: 
076:         /* Event0: match1 -> output0 high */
077:         sct_setup_event(0, 1, 0, SCT_STATE0 | SCT_MATCH_ONLY,
078:                         SCT_EV_OUT0_SET, 0);
079: 
080:         /* Event1: match2 -> output0 low */
081:         sct_setup_event(1, 2, 0, SCT_STATE0 | SCT_MATCH_ONLY,
082:                         SCT_EV_OUT0_CLR, 0);
083: 
084:         /* Start timer. */
085:         sct_start_counter();
086: }
087: 
088: static void mrt_setup(void)
089: {
090:         /* Enable MRT clock. */
091:         syscon_enable_clock(SYSCON_MRT);
092: 
093:         /* Set channel 0 to 'bus-stall mode'. */
094:         mrt_set_mode(MRT0, MRT_BUS_STALL);
095: 
096:         /* Set channel 1 to 'one-shot interrupt mode'. */
097:         mrt_set_mode(MRT1, MRT_ONE_SHOT);
098: }
099: 
100: static void acmp_setup(void)
101: {
102:         /* Enable ACMP power. */
103:         syscon_disable_power_down(SYSCON_ACMP_PD);
104: 
105:         /* Enable ACMP clock. */
106:         syscon_enable_clock(SYSCON_ACMP);
107: 
108:         /* Set up comparator. */
109:         acmp_init(ACMP_I1, ACMP_LADDER, ACMP_NONE, false, 0);
110: 
111:         /* Enable voltage ladder. */
112:         acmp_enable_ladder(false);
113: }
114: 
115: static void delay_us(int us)
116: {
117:         mrt_set_interval(MRT0, 24000000 / 1000000 * us);
118: }
119: 
120: static void delay_ms(int ms)
121: {
122:         mrt_set_interval(MRT1, 24000000 / 1000 * ms);
123: 
124:         while (!mrt_get_channel_status(MRT1, MRT_INT))
125:                 ;
126: 
127:         mrt_clear_channel_status(MRT1, MRT_INT);
128: }
129: 
130: int main(void)
131: {
132:         int i;
133: 
134:         clock_setup();
135:         gpio_setup();
136:         sct_setup();
137:         mrt_setup();
138:         acmp_setup();
139: 
140:         /* voltage ladder power-up settling time */
141:         delay_us(ACMP_TS_PU);
142: 
143:         while (1) {
144:                 for (i = 0; i < 32; i++) {
145:                         /* Switch voltage ladder. */
146:                         acmp_set_ladder(i);
147: 
148:                         /* switching settling time + propagation delay */
149:                         delay_us(ACMP_TS_SW + 1);
150: 
151:                         if (!acmp_get_status(ACMP_COMPSTAT))
152:                                 break;
153:                 }
154: 
155:                 sct_set_match_reload(2, 24000000 / 100 / 32 * i);
156: 
157:                 delay_ms(100);
158:         }
159: 
160:         return 0;
161: }
