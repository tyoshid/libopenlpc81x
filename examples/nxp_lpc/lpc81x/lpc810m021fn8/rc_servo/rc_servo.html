020: #include <syscon.h>
021: #include <gpio.h>
022: #include <sct.h>
023: 
024: #define PCLK    24000000
025: 
026: /* Set LPC810 to 24 MHz. */
027: static void clock_setup(void)
028: {
029:         /* Set up PLL (Fclkin = 12MHz (IRC), Fclkout (Main clock) = 24MHz) */
030:         syscon_enable_pll(SYSCON_IRC, 2, 4);
031: 
032:         /* Select PLL as system clock source. */
033:         syscon_set_system_clock(SYSCON_PLL_OUT, 1);
034: }
035: 
036: static void gpio_setup(void)
037: {
038:         /* Enable GPIO clock. */
039:         // syscon_enable_clock(SYSCON_GPIO);
040: 
041:         /* Enable switch matrix clock. */
042:         // syscon_enable_clock(SYSCON_SWM);
043: 
044:         /* Enable IOCON clock. */
045:         syscon_enable_clock(SYSCON_IOCON);
046: 
047:         /* Prevent the I2C pins from internally floating. */
048:         gpio_config(GPIO_OUTPUT, GPIO_IO, PIO0_10 | PIO0_11);
049:         gpio_clear(PIO0_10 | PIO0_11);
050: 
051:         /* Set PIO0_2 to 'SCT output 0'. */
052:         gpio_config(GPIO_CTOUT0, 0, PIO0_2);
053: 
054:         /* Set PIO0_3 to 'SCT input 0'. */
055:         gpio_config(GPIO_CTIN0, GPIO_HYST, PIO0_3);
056: }
057: 
058: static void sct_setup(void)
059: {
060:         /* Enable SCT clock. */
061:         syscon_enable_clock(SYSCON_SCT);
062: 
063:         /* Set counter to 32-bit. */
064:         sct_config(SCT_CLOCK_BUS, 0, SCT_UNIFY | SCT_AUTOLIMIT);
065: 
066:         /* Match0: limit */
067:         /* 50Hz */
068:         sct_set_match_and_reload(0, PCLK / 50 - 1);
069: 
070:         /* Match1: PWM low threshold (state 0) */
071:         sct_set_match_and_reload(1, 0);
072: 
073:         /* Match2: PWM high threshold (state 0) */
074:         /* pulse width: 1 msec */
075:         sct_set_match_and_reload(2, PCLK / 1000 - 1);
076: 
077:         /* Match3: PWM low threshold (state 1) */
078:         sct_set_match_and_reload(3, 0);
079: 
080:         /* Match4: PWM high threshold (state 1) */
081:         /* pulse width: 2 msec */
082:         sct_set_match_and_reload(4, PCLK / 500 - 1);
083: 
084:         /* Event0: match1 -> output0 low (state 0) */
085:         sct_setup_event(0, 1, 0, SCT_STATE0 | SCT_MATCH_ONLY,
086:                         SCT_EV_OUT0_CLR, 0);
087: 
088:         /* Event1: match2 -> output0 high (state 0) */
089:         sct_setup_event(1, 2, 0, SCT_STATE0 | SCT_MATCH_ONLY,
090:                         SCT_EV_OUT0_SET, 0);
091: 
092:         /* Event2: input0 (falling edge) -> state 1 */
093:         sct_setup_event(2, 0, 0, SCT_STATE0 | SCT_IO_ONLY | SCT_FALL,
094:                         SCT_EV_STATE_LOAD, 1);
095: 
096:         /* Event3: match3 -> output0 low (state 1) */
097:         sct_setup_event(3, 3, 0, SCT_STATE1 | SCT_MATCH_ONLY,
098:                         SCT_EV_OUT0_CLR, 0);
099: 
100:         /* Event4: match4 -> output0 high (state 1) */
101:         sct_setup_event(4, 4, 0, SCT_STATE1 | SCT_MATCH_ONLY,
102:                         SCT_EV_OUT0_SET, 0);
103: 
104:         /* Event5: input0 (falling edge) -> state 0 */
105:         sct_setup_event(5, 0, 0, SCT_STATE1 | SCT_IO_ONLY | SCT_FALL,
106:                         SCT_EV_STATE_LOAD, 0);
107: 
108:         /* Start timer. */
109:         sct_start_counter();
110: }
111: 
112: int main(void)
113: {
114:         clock_setup();
115:         gpio_setup();
116:         sct_setup();
117: 
118:         while (1)
119:                 ;
120: 
121:         return 0;
122: }
