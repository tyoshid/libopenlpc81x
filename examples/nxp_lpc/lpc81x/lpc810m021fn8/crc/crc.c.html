317: static void clock_setup(void)
318: {
319:         /* Set up PLL (Fclkin = 12MHz (IRC), Fclkout (Main clock) = 24MHz) */
320:         syscon_enable_pll(SYSCON_IRC, 2, 4);
321: 
322:         /* Select PLL as system clock source. */
323:         syscon_set_system_clock(SYSCON_PLL_OUT, 1);
324: 
325:         /* Enable USART clock (U_PCLK). */
326:         syscon_set_usart_clock(24000000, U_PCLK);
327: }

329: static void gpio_setup(void)
330: {
331:         /* Enable GPIO clock. */
332:         // syscon_enable_clock(SYSCON_GPIO);
333: 
334:         /* Enable switch matrix clock. */
335:         // syscon_enable_clock(SYSCON_SWM);
336: 
337:         /* Enable IOCON clock. */
338:         syscon_enable_clock(SYSCON_IOCON);
339: 
340:         /* Prevent the pins from internally floating. */
341:         gpio_config(GPIO_OUTPUT, GPIO_IO, PIO0_10 | PIO0_11);
342:         gpio_clear(PIO0_10 | PIO0_11);
343: 
344:         /* Set PIO0_2 and PIO0_3 to 'output push-pull' (LED). */
345:         gpio_config(GPIO_OUTPUT, 0, PIO0_2 | PIO0_3);
346: 
347:         /* Set PIO0_4 to 'USART0 TXD'. */
348:         gpio_config(GPIO_U0_TXD, GPIO_HYST, PIO0_4);
349: }

376: static void print_hex(int c)
377: {
378:         int d;
379: 
380:         d = (c >> 4) & 0xf;
381:         if (d > 9)
382:                 d += 'a' - 10;
383:         else
384:                 d += '0';
385:         usart_send_blocking(USART0, d);
386:         d = c & 0xf;
387:         if (d > 9)
388:                 d += 'a' - 10;
389:         else
390:                 d += '0';
391:         usart_send_blocking(USART0, d);
392: }
393: 
394: static void print_hex_half_word(int w)
395: {
396:         print_hex((w >> 8) & 0xff);
397:         print_hex(w & 0xff);
398: }
399: 
400: static void print_hex_word(int w)
401: {
402:         print_hex((w >> 24) & 0xff);
403:         print_hex((w >> 16) & 0xff);
404:         print_hex((w >> 8) & 0xff);
405:         print_hex(w & 0xff);
406: }

408: static void print_int(int d)
409: {
410:         bool zerosup;
411:         int i;
412: 
413:         if (d < 0) {
414:                 usart_send_blocking(USART0, '-');
415:                 d = -d;
416:         }
417:         zerosup = true;
418:         for (i = 1000000000; i > 0; i /= 10) {
419:                 if (d / i || i == 1 || !zerosup) {
420:                         usart_send_blocking(USART0, d / i + '0');
421:                         d %= i;
422:                         zerosup = false;
423:                 }
424:         }
425: }

427: static void xprintf(char *p, int arg0, int arg1, int arg2)
428: {
429:         int arg[3];
430:         int i;
431: 
432:         arg[0] = arg0;
433:         arg[1] = arg1;
434:         arg[2] = arg2;
435:         i = 0;
436:         while (*p != '\0') {
437:                 if (*p == '%') {
438:                         switch (*++p) {
439:                         case '%':
440:                                 usart_send_blocking(USART0, *p++);
441:                                 break;
442:                         case 'x':
443:                                 if (i < 3)
444:                                         print_hex(arg[i++]);
445:                                 p++;
446:                                 break;
447:                         case 'h':
448:                                 if (i < 3)
449:                                         print_hex_half_word(arg[i++]);
450:                                 p++;
451:                                 break;
452:                         case 'w':
453:                                 if (i < 3)
454:                                         print_hex_word(arg[i++]);
455:                                 p++;
456:                                 break;
457:                         case 'd':
458:                                 if (i < 3)
459:                                         print_int(arg[i++]);
460:                                 p++;
461:                                 break;
462:                         default:
463:                                 usart_send_blocking(USART0, '?');
464:                                 p++;
465:                                 break;
466:                         }
467:                 } else {
468:                         usart_send_blocking(USART0, *p++);
469:                 }
470:         }
471: }

492: static int bitreverse16(int data)
493: {
494:         int i;
495:         int r;
496: 
497:         r = 0;
498:         for (i = 0; i < 16; i++) {
499:                 if (data & (1 << i))
500:                         r |= 1 << (15 - i);
501:         }
502:         return r;
503: }
