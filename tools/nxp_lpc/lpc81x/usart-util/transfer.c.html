032: int upload(int fd, FILE *stream, int bytes)
033: {
034:         uint32_t a;
035:         int i;
036:         int n;
037:         int r;
038:         uint8_t buf[SECTOR_SIZE];
039: 
040:         a = FLASH_ADDRESS;
041:         i = 0;
042:         while (i < bytes) {
043:                 n = (bytes - i) > sizeof(buf) ? sizeof(buf) : bytes - i;
044: 
045:                 r = read_memory(fd, a, n, buf);
046:                 if (r)
047:                         return -1;
048:                 r = fwrite(buf, 1, n, stream);
049:                 if (r != n) {
050:                         fprintf(stderr, "fwrite() failed\n");
051:                         return -1;
052:                 }
053: 
054:                 if (debug)
055:                         printf("%08x\n", a);
056: 
057:                 a += n;
058:                 i += n;
059:         }
060: 
061:         printf("read %d bytes\n", bytes);
062:         return bytes;
063: }

065: int download(int fd, FILE *stream)
066: {
067:         uint32_t ramaddr;
068:         int ramsize;
069:         int sector;
070:         int bytes;
071:         uint32_t flashaddr;
072:         int r;
073:         uint8_t buf[SECTOR_SIZE];
074:         int n;
075:         int i;
076:         uint32_t w;
077:         int m;
078:         uint8_t flash[SECTOR_SIZE];
079: 
080:         ramaddr = SRAM_ADDRESS + RESERVE_SIZE;
081:         if (sram_size <= SECTOR_SIZE + RESERVE_SIZE)
082:                 ramsize = sram_size - RESERVE_SIZE;
083:         else
084:                 ramsize = SECTOR_SIZE;
085: 
086:         if (unlock(fd))
087:                 return -1;
088: 
089:         sector = 0;
090:         bytes = 0;
091:         flashaddr = FLASH_ADDRESS;
092:         while (!feof(stream)) {
093:                 r = fread(buf, 1, sizeof(buf), stream);
094:                 n = (r + PAGE_SIZE - 1) & ~(PAGE_SIZE - 1);
095:                 if (r < sizeof(buf)) {
096:                         if (ferror(stream)) {
097:                                 fprintf(stderr, "File read error\n");
098:                                 return -1;
099:                         }
100: 
101:                         if (r == 0)
102:                                 break;
103: 
104:                         for (i = r; i < sizeof(buf); i++)
105:                                 buf[i] = 0xff;
106:                 }
107:                 bytes += r;
108: 
109:                 /* Set checksum */
110:                 if (sector == 0) {
111:                         w = 0;
112:                         for (i = 0; i < 7; i++)
113:                                 w += buf[i * 4 + 3] << 24 |
114:                                         buf[i * 4 + 2] << 16 |
115:                                         buf[i * 4 + 1] << 8 |
116:                                         buf[i * 4];
117:                         w = -w;
118:                         buf[7 * 4] = w & 0xff;
119:                         buf[7 * 4 + 1] = w >> 8 & 0xff;
120:                         buf[7 * 4 + 2] = w >> 16 & 0xff;
121:                         buf[7 * 4 + 3] = w >> 24;
122: 
123:                         printf("Checksum = 0x%08x\n", w);
124:                 }
125: 
126:                 if (debug)
127:                         printf("- Blank check -\n");
128:                 /* Blank check */
129:                 r = blank_check_sectors(fd, sector, sector);
130:                 if (r == RESULT_SECTOR_NOT_BLANK) {
131:                         if (debug)
132:                                 printf("- Erase -\n");
133:                         /* Erase */
134:                         if (prepare_sectors(fd, sector, sector))
135:                                 return -1;
136: 
137:                         if (erase_sectors(fd, sector, sector))
138:                                 return -1;
139: 
140:                         /* Erase check */
141:                         if (blank_check_sectors(fd, sector, sector))
142:                                 return -1;
143:                 } else if (r) {
144:                         return -1;
145:                 }
146: 
147:                 if (debug)
148:                         printf("- Write data -\n");
149:                 /* Write data */
150:                 for (i = 0; i < n; i += m) {
151:                         if (n - i <= 64)
152:                                 m = 64; /* PAGE_SIZE */
153:                         else if (n - i <= 128)
154:                                 m = 128;
155:                         else if (ramsize <= 256)
156:                                 m = 128;
157:                         else if (n - i <= 256)
158:                                 m = 256;
159:                         else if (n - i <= 512)
160:                                 m = 512;
161:                         else
162:                                 m = 1024; /* SECTOR_SIZE */
163:                         if (write_to_ram(fd, ramaddr, m, &buf[i]))
164:                                 return -1;
165: 
166:                         if (prepare_sectors(fd, sector, sector))
167:                                 return -1;
168: 
169:                         if (copy_ram_to_flash(fd, flashaddr + i, ramaddr, m))
170:                                 return -1;
171:                 }
172: 
173:                 if (debug)
174:                         printf("- Verify data -\n");
175:                 /* Verify data */
176:                 if (read_memory(fd, flashaddr, n, flash))
177:                         return -1;
178: 
179:                 for (i = 0; i < n; i++) {
180:                         if (buf[i] != flash[i]) {
181:                                 fprintf(stderr, "Verify failed\n");
182:                                 return -1;
183:                         }
184:                 }
185: 
186:                 sector++;
187:                 flashaddr += n;
188:         }
189: 
190:         /* Check Code Read Protection */
191:         if (read_memory(fd, CRP, 4, flash))
192:                 return -1;
193: 
194:         w = flash[3] << 24 |
195:                 flash[2] << 16 |
196:                 flash[1] << 8 |
197:                 flash[0];
198:         if (debug)
199:                 printf("CRP = 0x%08x\n", w);
200: 
201:         switch (w) {
202:         case 0x12345678:
203:                 printf("CRP: CRP1\n");
204:                 break;
205:         case 0x87654321:
206:                 printf("CRP: CRP2\n");
207:                 break;
208:         case 0x43218765:
209:                 printf("CRP: CRP3\n");
210:                 break;
211:         case 0x4e697370:
212:                 printf("CRP: NO_ISP\n");
213:                 break;
214:         default:
215:                 printf("CRP: NONE\n");
216:                 break;
217:         }
218: 
219:         printf("wrote %d bytes\n", bytes);
220:         return bytes;
221: }
