037: static int com_write(int fd, char *s, int size)
038: {
039:         int i;
040:         int r;
041:         char d;
042: 
043:         if (echo_disable) {
044:                 if (write(fd, s, size) < 0) {
045:                         perror("write() failed");
046:                         return -1;
047:                 }
048:         } else {
049:                 for (i = 0; i < size; i++) {
050:                         r = write(fd, s, 1);
051:                         if (r < 0) {
052:                                 perror("write() failed");
053:                                 return -1;
054:                         }
055: 
056:                         /* Wait for echo */
057:                         r = read(fd, &d, 1);
058:                         if (r < 0) {
059:                                 perror("read() failed");
060:                                 return -1;
061:                         }
062:                         if (r != 1)
063:                                 return ERROR_TIMEOUT;
064:                         if (d != *s)
065:                                 return ERROR_INVALID_VALUE;
066: 
067:                         s++;
068:                 }
069:         }
070: 
071:         return 0;
072: }

154: /* ISP initialization */
155: int isp_init(int fd, int retry)
156: {
157:         int i;
158:         char buf[32];
159:         int r;
160: 
161:         for (i = 0; i < retry; i++) {
162:                 /* Send '?'(0x3F) */
163:                 buf[0] = '?';
164:                 if (write(fd, buf, 1) < 0) {
165:                         perror("write() failed");
166:                         return -1;
167:                 }
168: 
169:                 /* Wait for "Synchronized" */
170:                 r = com_gets(fd, buf, sizeof(buf));
171:                 if (r < 0)
172:                         return r;
173:                 if (!r && !strcmp(buf, "Synchronized\r\n")) {
174:                         break;
175:                 }
176:         }
177:         if (i >= retry) {
178:                 fprintf(stderr, "can't get \"Synchronized\"\n");
179:                 return -1;
180:         }
181: 
182:         if (debug)
183:                 printf("Synchronized\n");
184: 
185:         /* Send "Synchronized" */
186:         r = com_puts(fd, "Synchronized\r\n");
187:         if (r < 0)
188:                 return r;
189:         if (r) {
190:                 fprintf(stderr, "can't send \"Synchronized\"\n");
191:                 return r;
192:         }
193: 
194:         /* Wait for "OK" */
195:         r = com_gets(fd, buf, sizeof(buf));
196:         if (r < 0)
197:                 return r;
198:         if (r || strcmp(buf, "OK\r\n")) {
199:                 fprintf(stderr, "can't get \"OK\"\n");
200:                 if (r)
201:                         return r;
202:                 return ERROR_INVALID_VALUE;
203:         }
204: 
205:         if (debug)
206:                 printf("OK\n");
207: 
208:         /* Send clock frequency */
209:         r = com_puts(fd, "12000\r\n");
210:         if (r < 0)
211:                 return r;
212:         if (r) {
213:                 fprintf(stderr, "can't send clock frequency\n");
214:                 return r;
215:         }
216: 
217:         /* Wait for "OK" */
218:         r = com_gets(fd, buf, sizeof(buf));
219:         if (r < 0)
220:                 return r;
221:         if (r || strcmp(buf, "OK\r\n")) {
222:                 fprintf(stderr, "can't get \"OK\"\n");
223:                 if (r)
224:                         return r;
225:                 return ERROR_INVALID_VALUE;
226:         }
227: 
228:         return 0;
229: }
