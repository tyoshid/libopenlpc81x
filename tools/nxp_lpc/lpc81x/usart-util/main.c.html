023: #include <stdint.h>
024: #include <stdbool.h>

031: #include <termios.h>

034: #include "command.h"
035: #include "transfer.h"

037: #define RETRY 10
038: 
039: bool debug;
040: int sram_size = 1024;
041: 
042: static struct {
043:         char *name;
044:         speed_t speed;
045: } baud_table[] = {
046:         {"9600", B9600},
047:         {"19200", B19200},
048:         {"38400", B38400},
049:         {"57600", B57600},
050:         {"115200", B115200},
051:         {0, 0}
052: };
053: 
054: static struct {
055:         uint32_t pid;
056:         char *dev_name;
057:         int sram;
058: } device_table[] = {
059:         {PID_LPC810M021FN8, "LPC810M021FN8", 1024},
060:         {PID_LPC811M001JDH16, "LPC811M001JDH16", 2048},
061:         {PID_LPC812M101JDH16, "LPC812M101JDH16", 4096},
062:         {PID_LPC812M101JD20, "LPC812M101JD20", 4096},
063:         {PID_LPC812M101JDH20, "LPC812M101JDH20/JTB16", 4096},
064:         {0, NULL, 0}
065: };
066: 
067: static void usage(char *prog)
068: {
069:         printf("Usage: %s [options]\n", prog);
070:         printf("  -h\t\tPrint this message\n");
071:         printf("  -v\t\tPrint verbose debug statements\n");
072:         printf("  -d <dev>\tSpecify USART device (default: /dev/ttyUSB0)\n");
073:         printf("  -b <baud>\tSpecify baud rate (default: 115200)\n");
074:         printf("  -t <bytes>\tSpecify the number of upload transfer bytes\n");
075:         printf("  -U <file>\tRead firmware from device into file\n");
076:         printf("  -D <file>\tWrite firmware from file into device\n");
077: }
078: 
079: int main(int argc, char *argv[])
080: {
081:         int opt;
082:         int i;
083:         speed_t baudrate = B115200;
084:         bool download_flag = false;
085:         char *fname = NULL;
086:         char *usart = "/dev/ttyUSB0";
087:         int size = 256;
088:         bool upload_flag = false;
089:         FILE *stream;
090:         int fd;
091:         struct termios oldtio;
092:         struct termios newtio;
093:         uint8_t version[2];
094:         uint32_t pid;
095: 
096:         while ((opt = getopt(argc, argv, "b:D:d:ht:U:v")) != -1) {
097:                 switch (opt) {
098:                 case 'b':
099:                         for (i = 0; baud_table[i].name; i++) {
100:                                 if (strcmp(optarg, baud_table[i].name) == 0) {
101:                                         baudrate = baud_table[i].speed;
102:                                         break;
103:                                 }
104:                         }
105:                         if (!baud_table[i].name) {
106:                                 fprintf(stderr, "Invalid baud rate (%s).\n",
107:                                         optarg);
108:                                 return 1;
109:                         }
110:                         break;
111:                 case 'D':
112:                         download_flag = true;
113:                         fname = optarg;
114:                         break;
115:                 case 'd':
116:                         usart = optarg;
117:                         break;
118:                 case 'h':
119:                         usage(argv[0]);
120:                         return 0;
121:                 case 't':
122:                         size = atoi(optarg);
123:                         break;
124:                 case 'U':
125:                         upload_flag = true;
126:                         fname = optarg;
127:                         break;
128:                 case 'v':
129:                         debug = true;
130:                         break;
131:                 default:
132:                         usage(argv[0]);
133:                         return 1;
134:                 }
135:         }
136: 
137:         if ((download_flag && upload_flag) ||
138:             (!download_flag && !upload_flag)) {
139:                 fprintf(stderr, "You need to specify one of -D or -U\n");
140:                 return 1;
141:         }
142: 
143:         /* Open data file. */
144:         if (download_flag)
145:                 stream = fopen(fname, "r");
146:         else
147:                 stream = fopen(fname, "w");
148:         if (stream == NULL) {
149:                 perror(fname);
150:                 return 1;
151:         }
152: 
153:         /* Open serial port. */
154:         fd = open(usart, O_RDWR | O_NOCTTY);
155:         if (fd < 0) {
156:                 perror(usart);
157:                 fclose(stream);
158:                 return 1;
159:         }
160: 
161:         if (tcgetattr(fd, &oldtio)) {
162:                 perror("tcgetattr() failed");
163:                 close(fd);
164:                 fclose(stream);
165:                 return 1;
166:         }
167: 
168:         bzero(&newtio, sizeof(newtio));
169:         newtio.c_cflag = baudrate | CS8 | CLOCAL | CREAD;
170:         newtio.c_iflag = IGNPAR;
171:         newtio.c_oflag = 0;
172: 
173:         newtio.c_lflag = 0;
174: 
175:         newtio.c_cc[VTIME] = 5;
176:         newtio.c_cc[VMIN] = 0;
177: 
178:         if (tcflush(fd, TCIFLUSH)) {
179:                 perror("tcflush() failed");
180:                 close(fd);
181:                 fclose(stream);
182:                 return 1;
183:         }
184:         if (tcsetattr(fd, TCSANOW, &newtio)) {
185:                 perror("tcsetattr() failed");
186:                 close(fd);
187:                 fclose(stream);
188:                 return 1;
189:         }
190: 
191:         /* ISP initialization */
192:         if (isp_init(fd, RETRY))
193:                 goto ioerror;
194: 
195:         /* Echo off */
196:         if (echo(fd, 0))
197:                 goto ioerror;
198: 
199:         /* Read ISP version */
200:         if (read_boot_code_version(fd, version))
201:                 goto ioerror;
202:         printf("ISP version %d.%d\n", version[1], version[0]);
203: 
204:         /* Read PID */
205:         if (read_part_id(fd, &pid))
206:                 goto ioerror;
207:         printf("PID 0x%x  ", pid);
208:         for (i = 0; device_table[i].dev_name; i++) {
209:                 if (device_table[i].pid == pid) {
210:                         printf("%s\n", device_table[i].dev_name);
211:                         sram_size = device_table[i].sram;
212:                         break;
213:                 }
214:         }
215:         if (!device_table[i].dev_name)
216:                 printf("unknown device\n");
217: 
218:         /* Transfer data. */
219:         if (upload_flag) {
220:                 if (upload(fd, stream, size) < 0)
221:                         goto ioerror;
222:         }
223:         if (download_flag) {
224:                 if (download(fd, stream) < 0)
225:                         goto ioerror;
226:         }
227: 
228:         /* Close serial port. */
229:         if (tcsetattr(fd, TCSANOW, &oldtio)) {
230:                 perror("tcsetattr() failed");
231:                 close(fd);
232:                 fclose(stream);
233:                 return 1;
234:         }
235:         if (close(fd)) {
236:                 perror("close() failed");
237:                 fclose(stream);
238:                 return 1;
239:         }
240: 
241:         /* Close data file. */
242:         if (fclose(stream)) {
243:                 perror("fclose() failed");
244:                 return 1;
245:         }
246: 
247:         return 0;
248: 
249: ioerror:
250:         tcsetattr(fd, TCSANOW, &oldtio);
251:         close(fd);
252:         fclose(stream);
253:         return 1;
254: }
